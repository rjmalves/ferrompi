1//! MPI message status information.
2//!
3//! This module provides the [`Status`] struct returned by probe operations,
4//! containing metadata about a message without actually receiving it.
5
6/// Information about a probed or received MPI message.
7///
8/// Returned by [`Communicator::probe`](crate::Communicator::probe) and
9/// [`Communicator::iprobe`](crate::Communicator::iprobe) to describe an
10/// incoming message without consuming it.
11///
12/// # Example
13///
14/// ```no_run
15/// # use ferrompi::Mpi;
16/// let mpi = Mpi::init().unwrap();
17/// let world = mpi.world();
18///
19/// // Blocking probe for any f64 message
20/// let status = world.probe::<f64>(-1, -1).unwrap();
21/// println!("Message from rank {} with tag {}, {} elements",
22///          status.source, status.tag, status.count);
23/// ```
24#[derive(Debug, Clone)]
25pub struct Status {
26    /// Source rank of the message.
27    pub source: i32,
28    /// Tag of the message.
29    pub tag: i32,
30    /// Number of elements in the message (determined via `MPI_Get_count`).
31    pub count: i64,
32}

34#[cfg(test)]
35mod tests {

37
38    #[test]
39    fn status_clone() {
40        let status = Status {
41            source: 3,
42            tag: 42,
43            count: 100,
44        };
45        let cloned = status.clone();
46        assert_eq!(cloned.source, 3);
47        assert_eq!(cloned.tag, 42);
48        assert_eq!(cloned.count, 100);
49    }
50
51    #[test]
52    fn status_debug() {
53        let status = Status {
54            source: 0,
55            tag: 1,
56            count: 50,
57        };
58        let debug = format!("{:?}", status);
59        assert!(debug.contains("source"));
60        assert!(debug.contains("tag"));
61        assert!(debug.contains("count"));
62    }
