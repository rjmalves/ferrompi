1//! MPI Group handle for rank-set operations.
2//!
3//! [`Group`] wraps an `MPI_Group` handle via the C-side group table and
4//! provides methods for querying group size and rank. Use
5//! [`Communicator::group`](crate::Communicator::group) to obtain a `Group`
6//! from an existing communicator, then call [`include`](Group::include) or
7//! [`exclude`](Group::exclude) to derive sub-groups.
8//!
9//! # Example
10//!
11//! ```no_run
12//! use ferrompi::Mpi;
13//!
14//! let mpi = Mpi::init().unwrap();
15//! let world = mpi.world();
16//!
17//! let world_group = world.group().unwrap();
18//! assert_eq!(world_group.size().unwrap(), world.size());
19//!
20//! // Include only ranks 0 and 2
21//! let sub = world_group.include(&[0, 2]).unwrap();
22//! assert_eq!(sub.size().unwrap(), 2);
23//! ```
24

28/// Outcome of [`Group::compare`].
29///
30/// Mirrors `MPI_Group_compare`'s three possible result codes:
31/// `MPI_IDENT`, `MPI_SIMILAR`, `MPI_UNEQUAL`.
32///
33/// The discriminant values (0, 1, 2) are ferrompi-stable and do **not**
34/// correspond directly to the MPI implementation's integer constants, which
35/// differ between MPICH and Open MPI. Normalisation is performed in the C shim
36/// so that this enum can use `#[repr(i32)]` with fixed discriminants.
37///
38/// # Example
39///
40/// ```no_run
41/// use ferrompi::{GroupComparison, Mpi};
42///
43/// let mpi = Mpi::init().unwrap();
44/// let world = mpi.world();
45/// let gw = world.group().unwrap();
46///
47/// // Same object → Identical
48/// assert_eq!(gw.compare(&gw).unwrap(), GroupComparison::Identical);
49/// ```
50#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
51#[repr(i32)]
52pub enum GroupComparison {
53    /// `MPI_IDENT` — same group object (identity, not just rank equality).
54    Identical = 0,
55    /// `MPI_SIMILAR` — same rank set, possibly different ordering.
56    Similar = 1,
57    /// `MPI_UNEQUAL` — different rank sets.
58    Unequal = 2,
59}

61/// Inclusive rank range `[first, last]` with positive stride.
62///
63/// Used as input to [`Group::range_include`] and
64/// [`Group::range_exclude`] for compact specification of arithmetic
65/// progressions over rank ids.
66///
67/// Maps to a single row of the `ranges[3][N]` array argument of
68/// `MPI_Group_range_incl` / `MPI_Group_range_excl`.
69///
70/// # Example
71///
72/// ```no_run
73/// use ferrompi::{Mpi, RankRange};
74///
75/// let mpi = Mpi::init().unwrap();
76/// let world = mpi.world();
77/// let gw = world.group().unwrap();
78///
79/// // Include ranks 0, 2 (every other rank starting from 0, up to 3)
80/// let sub = gw.range_include(&[RankRange { first: 0, last: 3, stride: 2 }]).unwrap();
81/// assert_eq!(sub.size().unwrap(), 2);
82/// ```
83#[derive(Debug, Clone, Copy, PartialEq, Eq)]
84pub struct RankRange {
85    /// First rank in the range (inclusive).
86    pub first: i32,
87    /// Last rank in the range (inclusive).
88    pub last: i32,
89    /// Step between consecutive ranks (must be positive).
90    pub stride: i32,
91}

93/// An MPI group handle.
94///
95/// This type wraps an `MPI_Group` handle with RAII semantics: the underlying
96/// MPI group object is freed automatically when the `Group` is dropped,
97/// unless it is slot 0 (reserved for `MPI_GROUP_EMPTY`).
98///
99/// Obtain a `Group` from a communicator via
100/// [`Communicator::group`](crate::Communicator::group), then use
101/// [`include`](Self::include) or [`exclude`](Self::exclude) to derive
102/// sub-groups.
103///
104/// # MPI_UNDEFINED
105///
106/// [`rank`](Self::rank) returns `-1` when the calling process is not a
107/// member of the group (`MPI_UNDEFINED` in the MPI standard). Consumers
108/// must handle this sentinel explicitly.
109pub struct Group {
110    pub(crate) handle: i32,
111}
112
113// SAFETY: Group handles are integer indices into a C-side table, identical
114// in nature to Communicator handles. The C MPI library manages its own
115// thread safety based on the thread level requested via MPI_Init_thread.
116// Sending a Group to another thread is safe under the same conditions as
117// Communicator (see src/comm/mod.rs); users must ensure adequate thread
118// support and serialize access when using ThreadLevel::Serialized.
119unsafe impl Send for Group {}
120unsafe impl Sync for Group {}

123    /// Get the number of processes in this group.
124    ///
125    /// # Errors
126    ///
127    /// Returns an error if the underlying MPI call fails.
128    ///
129    /// # Example
130    ///
131    /// ```no_run
132    /// use ferrompi::Mpi;
133    ///
134    /// let mpi = Mpi::init().unwrap();
135    /// let world = mpi.world();
136    /// let g = world.group().unwrap();
137    /// assert_eq!(g.size().unwrap(), world.size());

139    pub fn size(&self) -> Result<i32> {
140        let mut size: i32 = 0;
141        // SAFETY: self.handle is owned by this Group and was allocated by the C-side group table.
142        let ret = unsafe { ffi::ferrompi_group_size(self.handle, &mut size) };
143        Error::check_with_op(ret, "group_size")?;
144        Ok(size)
145    }

179    /// Return the normalised `MPI_UNDEFINED` sentinel value (`-1`).
180    ///
181    /// [`rank`](Self::rank) returns `-1` when the calling process is not a
182    /// member of the group. The C shim normalises the implementation-defined
183    /// `MPI_UNDEFINED` constant (e.g. `-32766` on MPICH) to `-1` before
184    /// returning, so this method always returns `-1` regardless of the
185    /// underlying MPI implementation.
186    ///
187    /// This function does not require an active MPI session.
188    ///
189    /// # Example
190    ///
191    /// ```no_run
192    /// use ferrompi::{Group, Mpi};
193    ///
194    /// let mpi = Mpi::init().unwrap();
195    /// let world = mpi.world();
196    /// let sub = world.group().unwrap().include(&[0, 2]).unwrap();
197    /// let r = sub.rank().unwrap();
198    /// if r == Group::undefined() {
199    ///     println!("not a member of this sub-group");
200    /// }
201    /// ```
202    pub fn undefined() -> i32 {
203        // SAFETY: ferrompi_mpi_undefined is a pure query with no pointer arguments.
204        unsafe { ffi::ferrompi_mpi_undefined() }
205    }

207    /// Create a new group containing only the specified ranks from this group.
208    ///
209    /// The resulting group contains the processes from this group whose ranks
210    /// appear in the `ranks` slice, in the order given.
211    ///
212    /// # Arguments
213    ///
214    /// * `ranks` - Slice of rank indices from this group to include.
215    ///
216    /// # Errors
217    ///
218    /// Returns an error if any rank in `ranks` is out of range for this group,
219    /// or if the C-side group table is full.
220    ///
221    /// # Example
222    ///
223    /// ```no_run
224    /// use ferrompi::Mpi;
225    ///
226    /// let mpi = Mpi::init().unwrap();
227    /// let world = mpi.world();
228    /// let sub = world.group().unwrap().include(&[0, 2]).unwrap();
229    /// assert_eq!(sub.size().unwrap(), 2);
230    /// ```
231    pub fn include(&self, ranks: &[i32]) -> Result<Group> {
232        let mut new_handle: i32 = 0;
233        // SAFETY: self.handle is owned; ranks.as_ptr() is valid for ranks.len() elements
234        // for the duration of this call.
235        let ret = unsafe {
236            ffi::ferrompi_group_incl(
237                self.handle,
238                ranks.len() as i32,
239                ranks.as_ptr(),
240                &mut new_handle,
241            )
242        };
243        Error::check_with_op(ret, "group_incl")?;
244        Ok(Group { handle: new_handle })
245    }

248    ///
249    /// The resulting group contains all ranks from this group (in their order),
250    /// followed by ranks from `other` that are not already in this group.
251    ///
252    /// # Errors
253    ///
254    /// Returns an error if the underlying MPI call fails or the C-side group

257    /// # Example

262    /// let mpi = Mpi::init().unwrap();

264    /// let gw = world.group().unwrap();
265    /// let g1 = gw.include(&[0, 1]).unwrap();
266    /// let g2 = gw.include(&[1, 2]).unwrap();
267    /// assert_eq!(g1.union(&g2).unwrap().size().unwrap(), 3);
268    /// ```
269    pub fn union(&self, other: &Group) -> Result<Group> {
270        let mut h: i32 = -1;
271        // SAFETY: both self.handle and other.handle are owned by their respective Groups.
272        let ret = unsafe { ffi::ferrompi_group_union(self.handle, other.handle, &mut h) };
273        Error::check_with_op(ret, "group_union")?;
274        Ok(Group { handle: h })
275    }

336    /// Compare this group against another.
337    ///
338    /// Returns:
339    /// - [`GroupComparison::Identical`] if `self` and `other` are the same MPI
340    ///   group object (object identity, not just rank equality).
341    /// - [`GroupComparison::Similar`] if both groups contain the same set of
342    ///   processes but in a different order.
343    /// - [`GroupComparison::Unequal`] if the rank sets differ.
344    ///
345    /// # Errors
346    ///
347    /// Returns an error if the underlying `MPI_Group_compare` call fails.
348    /// An unexpected return value from MPI (which should never happen on a
349    /// conforming implementation) is surfaced via `MPI_ERR_INTERN` from the
350    /// C shim and maps to [`Error::Mpi`] with class `Intern`.
351    ///
352    /// # Example
353    ///
354    /// ```no_run
355    /// use ferrompi::{GroupComparison, Mpi};
356    ///
357    /// let mpi = Mpi::init().unwrap();
358    /// let world = mpi.world();
359    /// let gw = world.group().unwrap();
360    /// let g1 = gw.include(&[0, 1, 2]).unwrap();
361    /// let g2 = gw.include(&[0, 1, 2]).unwrap();
362    ///
363    /// // Same rank set, different objects → Similar
364    /// assert_eq!(g1.compare(&g2).unwrap(), GroupComparison::Similar);
365    ///
366    /// // Same object → Identical
367    /// assert_eq!(gw.compare(&gw).unwrap(), GroupComparison::Identical);
368    /// ```
369    pub fn compare(&self, other: &Group) -> Result<GroupComparison> {
370        let mut result: i32 = -1;
371        // SAFETY: both self.handle and other.handle are owned by their respective Groups.
372        let ret = unsafe { ffi::ferrompi_group_compare(self.handle, other.handle, &mut result) };
373        Error::check_with_op(ret, "group_compare")?;
374        match result {
375            0 => Ok(GroupComparison::Identical),
376            1 => Ok(GroupComparison::Similar),
377            2 => Ok(GroupComparison::Unequal),
378            other => Err(Error::Internal(format!(
379                "ferrompi_group_compare returned unexpected result {other}"
380            ))),
381        }
382    }

385    /// in the `ranks` slice.
386    ///
387    /// # Arguments
388    ///
389    /// * `ranks` - Slice of rank indices from this group to exclude.
390    ///
391    /// # Errors
392    ///
393    /// Returns an error if any rank in `ranks` is out of range for this group,

409        // for the duration of this call.

412                self.handle,
413                ranks.len() as i32,
414                ranks.as_ptr(),
415                &mut new_handle,
416            )
417        };
418        Error::check_with_op(ret, "group_excl")?;
419        Ok(Group { handle: new_handle })
420    }

422    /// Create a new group containing the union of all rank progressions
423    /// described by the `ranges` triples.
424    ///
425    /// Each [`RankRange`] specifies an arithmetic progression
426    /// `first, first+stride, ..., last` of ranks to include from this group.
427    /// The result group contains those ranks in the order they appear across
428    /// the triples.
429    ///
430    /// An empty `ranges` slice produces an empty group (`size() == 0`).
431    ///
432    /// # Errors
433    ///
434    /// Returns an error if MPI validation rejects any triple (e.g., negative
435    /// or zero stride, `first > last` with positive stride), or if the
436    /// C-side group table is full.
437    ///
438    /// # Example
439    ///
440    /// ```no_run
441    /// use ferrompi::{Mpi, RankRange};
442    ///
443    /// let mpi = Mpi::init().unwrap();
444    /// let world = mpi.world();
445    /// let gw = world.group().unwrap();
446    ///
447    /// // Include ranks 0, 1, 2
448    /// let sub = gw.range_include(&[RankRange { first: 0, last: 2, stride: 1 }]).unwrap();
449    /// assert_eq!(sub.size().unwrap(), 3);
450    /// ```
451    pub fn range_include(&self, ranges: &[RankRange]) -> Result<Group> {
452        let mut flat: Vec<i32> = Vec::with_capacity(3 * ranges.len());
453        for r in ranges {
454            flat.push(r.first);
455            flat.push(r.last);
456            flat.push(r.stride);
457        }
458        let mut h: i32 = -1;
459        // SAFETY: self.handle is owned; flat is a contiguous Vec<i32> with 3*ranges.len() elements
460        // and outlives this call.
461        let ret = unsafe {
462            ffi::ferrompi_group_range_incl(self.handle, ranges.len() as i32, flat.as_ptr(), &mut h)
463        };
464        Error::check_with_op(ret, "group_range_incl")?;
465        Ok(Group { handle: h })
466    }

502        }

504        // SAFETY: self.handle is owned; flat is a contiguous Vec<i32> with 3*ranges.len() elements
505        // and outlives this call.
506        let ret = unsafe {
507            ffi::ferrompi_group_range_excl(self.handle, ranges.len() as i32, flat.as_ptr(), &mut h)

513    /// Translate `ranks` from `self`'s rank space into `other`'s rank space.
514    ///
515    /// Returns a `Vec<Option<i32>>` of the same length as `ranks`:
516    /// `Some(rank_in_other)` for ranks present in `other`, `None` for ranks
517    /// not present in `other` (`MPI_UNDEFINED` entries normalised by the C shim
518    /// to `-1` before Rust sees them).
519    ///
520    /// # Arguments
521    ///
522    /// * `ranks` — Input ranks, all referring to `self`'s rank space. Values may
523    ///   be any non-negative `i32`; out-of-range values surface as `MPI_ERR_RANK`
524    ///   from MPI.
525    /// * `other` — Target rank space.
526    ///
527    /// # Errors
528    ///
529    /// Returns an error if the underlying `MPI_Group_translate_ranks` call fails
530    /// (e.g. `MPI_ERR_RANK` for an out-of-range input rank).
531    ///
532    /// # Example
533    ///
534    /// ```no_run
535    /// use ferrompi::Mpi;
536    ///
537    /// let mpi = Mpi::init().unwrap();
538    /// let world = mpi.world();
539    /// let gw = world.group().unwrap();
540    /// let gsub = gw.include(&[1, 3]).unwrap();
541    ///
542    /// // World ranks 0 and 2 are not in gsub → None; ranks 1 and 3 map to 0 and 1.
543    /// let translated = gw.translate_ranks(&[0, 1, 2, 3], &gsub).unwrap();
544    /// assert_eq!(translated, vec![None, Some(0), None, Some(1)]);
545    /// ```
546    pub fn translate_ranks(&self, ranks: &[i32], other: &Group) -> Result<Vec<Option<i32>>> {
547        if ranks.is_empty() {
548            return Ok(vec![]);
549        }
550        let mut out: Vec<i32> = vec![-1; ranks.len()];
551        // SAFETY: self.handle and other.handle are owned; ranks.as_ptr() and out.as_mut_ptr() are
552        // valid for ranks.len() elements each with no aliasing between them.
553        let ret = unsafe {
554            ffi::ferrompi_group_translate_ranks(
555                self.handle,
556                ranks.len() as i32,
557                ranks.as_ptr(),
558                other.handle,
559                out.as_mut_ptr(),
560            )
561        };
562        Error::check_with_op(ret, "group_translate_ranks")?;
563        Ok(out
564            .into_iter()
565            .map(|r| if r == -1 { None } else { Some(r) })
566            .collect())
567    }

570impl Drop for Group {
571    fn drop(&mut self) {
572        // Slot 0 is reserved for MPI_GROUP_EMPTY and must not be freed.
573        if self.handle > 0 {
574            // SAFETY: self.handle is owned and valid; Drop runs exactly once, so no double-free.
575            unsafe { ffi::ferrompi_group_free(self.handle) };
576        }
577    }
578}

655    /// Empty-input fast path: translate_ranks with an empty slice must return
656    /// Ok(vec![]) without calling FFI (which would require an active MPI session).
657    ///
658    /// This test is safe to run without `mpiexec` because the wrapper
659    /// short-circuits on `ranks.is_empty()` before touching any MPI handle.
660    #[test]
661    fn translate_ranks_empty_input_returns_empty_vec() {
662        let g = Group { handle: 0 };
663        let result = g
664            .translate_ranks(&[], &Group { handle: 0 })
665            .expect("translate_ranks with empty slice must succeed");
666        assert!(result.is_empty(), "expected empty Vec, got {result:?}");
667        // Prevent Drop from calling ferrompi_group_free on handle 0 twice.
668        // The guard in Drop already skips handle 0, but forget is explicit
669        // about our intent here.
670        std::mem::forget(g);
671    }
