7impl Communicator {
8    /// Get the processor name for this process.
9    pub fn processor_name(&self) -> Result<String> {
10        let mut buf = [0u8; 256];
11        let mut len: i32 = 0;
12        let ret = unsafe {
13            ffi::ferrompi_get_processor_name(buf.as_mut_ptr().cast::<std::ffi::c_char>(), &mut len)
14        };
15        Error::check_with_op(ret, "get_processor_name")?;
16        let len = (len.max(0) as usize).min(buf.len());
17        let s = std::str::from_utf8(&buf[..len])
18            .map_err(|_| Error::Internal("Invalid UTF-8 in processor name".into()))?;
19        Ok(s.to_string())
20    }

22    /// Gather topology information from all ranks in this communicator.
23    ///
24    /// This is a **collective operation** — all ranks must call it.
25    /// Every rank receives the complete [`TopologyInfo`], which includes the
26    /// rank-to-host mapping, MPI version metadata, and (with the `numa`
27    /// feature) SLURM job information.
28    ///
29    /// [`TopologyInfo`]: crate::TopologyInfo
30    ///
31    /// # Example
32    ///
33    /// ```no_run
34    /// use ferrompi::Mpi;
35    ///
36    /// let mpi = Mpi::init().unwrap();
37    /// let world = mpi.world();
38    /// let topo = world.topology(&mpi).unwrap();
39    /// if world.rank() == 0 {
40    ///     println!("{topo}");
41    /// }
42    /// ```
43    pub fn topology(&self, mpi: &crate::Mpi) -> Result<crate::TopologyInfo> {
44        crate::topology::gather_topology(self, mpi)
45    }

47    /// Duplicate this communicator.
48    pub fn duplicate(&self) -> Result<Self> {
49        let mut new_handle: i32 = 0;
50        let ret = unsafe { ffi::ferrompi_comm_dup(self.handle, &mut new_handle) };
51        Error::check_with_op(ret, "comm_dup")?;
52        Self::from_handle(new_handle)
53    }

55    /// Split this communicator into sub-communicators based on color and key.
56    ///
57    /// Processes with the same `color` are placed in the same new communicator.
58    /// The `key` controls the rank ordering within the new communicator.
59    ///
60    /// Returns `None` if this process used [`Communicator::UNDEFINED`] as color.
61    ///
62    /// # Example
63    ///
64    /// ```no_run
65    /// use ferrompi::Mpi;
66    ///
67    /// let mpi = Mpi::init().unwrap();
68    /// let world = mpi.world();
69    /// let color = world.rank() % 2; // Even/odd split
70    /// if let Some(sub) = world.split(color, world.rank()).unwrap() {
71    ///     println!("Rank {} in sub-communicator of size {}", sub.rank(), sub.size());
72    /// }
73    /// ```
74    pub fn split(&self, color: i32, key: i32) -> Result<Option<Communicator>> {
75        let mut new_handle: i32 = 0;
76        let ret = unsafe { ffi::ferrompi_comm_split(self.handle, color, key, &mut new_handle) };
77        Error::check_with_op(ret, "comm_split")?;
78        if new_handle < 0 {
79            Ok(None)
80        } else {
81            Self::from_handle(new_handle).map(Some)
82        }
83    }

85    /// Split this communicator by type.
86    ///
87    /// Processes that share the same resource (determined by `split_type`) are
88    /// placed in the same new communicator. The `key` controls the rank ordering
89    /// within the new communicator.
90    ///
91    /// Returns `None` if MPI returns `MPI_COMM_NULL` for this process.
92    ///
93    /// # Example
94    ///
95    /// ```no_run
96    /// use ferrompi::{Mpi, SplitType};
97    ///
98    /// let mpi = Mpi::init().unwrap();
99    /// let world = mpi.world();
100    /// if let Some(node) = world.split_type(SplitType::Shared, world.rank()).unwrap() {
101    ///     println!("Node has {} processes", node.size());
102    /// }
103    /// ```
104    pub fn split_type(&self, split_type: SplitType, key: i32) -> Result<Option<Communicator>> {
105        let mut new_handle: i32 = 0;
106        let ret = unsafe {
107            ffi::ferrompi_comm_split_type(self.handle, split_type as i32, key, &mut new_handle)
108        };
109        Error::check_with_op(ret, "comm_split_type")?;
110        if new_handle < 0 {
111            Ok(None)
112        } else {
113            Self::from_handle(new_handle).map(Some)
114        }
115    }

117    /// Create a communicator containing only processes that share memory.
118    ///
119    /// This is equivalent to `split_type(SplitType::Shared, self.rank())`.
120    /// All processes on the same physical node will be in the same communicator.
121    ///
122    /// # Errors
123    ///
124    /// Returns [`Error::Internal`] if MPI unexpectedly returns a null communicator,
125    /// which should not happen for `MPI_COMM_TYPE_SHARED` under normal conditions.
126    ///
127    /// # Example
128    ///
129    /// ```no_run
130    /// use ferrompi::Mpi;
131    ///
132    /// let mpi = Mpi::init().unwrap();
133    /// let world = mpi.world();
134    /// let node = world.split_shared().unwrap();
135    /// println!("Node has {} processes, I am local rank {}", node.size(), node.rank());
136    /// ```
137    pub fn split_shared(&self) -> Result<Communicator> {
138        self.split_type(SplitType::Shared, self.rank())?
139            .ok_or_else(|| Error::Internal("split_shared returned null communicator".into()))
140    }

142    /// Abort MPI execution across all processes in this communicator.
143    ///
144    /// This function terminates all processes associated with the communicator.
145    /// It calls `MPI_Abort` and then exits the process. This function never
146    /// returns.
147    ///
148    /// # Arguments
149    ///
150    /// * `errorcode` - Error code to return to the invoking environment
151    ///
152    /// # Example
153    ///
154    /// ```no_run
155    /// # use ferrompi::Mpi;
156    /// # let mpi = Mpi::init().unwrap();
157    /// # let world = mpi.world();
158    /// if world.rank() == 0 {
159    ///     // Fatal error detected, abort all processes
160    ///     world.abort(1);
161    /// }
162    /// ```
163    ///
164    /// # Fallback Behavior
165    ///
166    /// In the standard case `MPI_Abort` terminates the entire process
167    /// group and this function never returns. If `MPI_Abort` itself
168    /// returns (non-standard but observed in some implementations during
169    /// internal failures), this function falls back to
170    /// [`std::process::abort`], which raises `SIGABRT` and terminates
171    /// the current process immediately. The original `errorcode` is lost
172    /// in that fallback path because `process::abort` does not accept an
173    /// exit code; the signal code (usually 134 = 128 + 6) is the best
174    /// evidence the caller has that the fallback triggered.
175    pub fn abort(&self, errorcode: i32) -> ! {
176        // SAFETY: ferrompi_abort delegates to MPI_Abort, which is
177        // defined to terminate all processes in the communicator and
178        // never return. This call is safe to make with any valid
179        // communicator handle; self.handle is always valid because
180        // Communicator is only constructed through safe methods that
181        // guarantee handle validity.
182        unsafe { ffi::ferrompi_abort(self.handle, errorcode) };
183
184        // Defense in depth: the MPI standard says MPI_Abort "should"
185        // terminate all processes but does not strictly guarantee the
186        // calling process aborts before return. If MPI_Abort ever
187        // returns (non-standard implementation behavior), we must still
188        // honor the `-> !` contract. std::process::abort() raises
189        // SIGABRT and is guaranteed to diverge without running
190        // destructors — which is the right outcome, because any
191        // destructor that touches MPI state after a failed MPI_Abort
192        // has undefined behavior.
193        std::process::abort()
194    }
