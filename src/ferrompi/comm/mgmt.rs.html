9    /// Get the processor name for this process.
10    pub fn processor_name(&self) -> Result<String> {
11        let mut buf = [0u8; 256];
12        let mut len: i32 = 0;
13        // SAFETY: buf is a 256-byte stack array, exclusively writable; the C shim writes at most
14        // MPI_MAX_PROCESSOR_NAME bytes and sets len to the actual length.
15        let ret = unsafe {
16            ffi::ferrompi_get_processor_name(buf.as_mut_ptr().cast::<std::ffi::c_char>(), &mut len)
17        };
18        Error::check_with_op(ret, "get_processor_name")?;
19        let len = (len.max(0) as usize).min(buf.len());
20        let s = std::str::from_utf8(&buf[..len])
21            .map_err(|_| Error::Internal("Invalid UTF-8 in processor name".into()))?;
22        Ok(s.to_string())
23    }

25    /// Gather topology information from all ranks in this communicator.
26    ///
27    /// This is a **collective operation** — all ranks must call it.
28    /// Every rank receives the complete [`TopologyInfo`], which includes the
29    /// rank-to-host mapping, MPI version metadata, and (with the `numa`
30    /// feature) SLURM job information.
31    ///
32    /// [`TopologyInfo`]: crate::TopologyInfo
33    ///
34    /// # Example
35    ///
36    /// ```no_run
37    /// use ferrompi::Mpi;
38    ///
39    /// let mpi = Mpi::init().unwrap();
40    /// let world = mpi.world();
41    /// let topo = world.topology(&mpi).unwrap();
42    /// if world.rank() == 0 {
43    ///     println!("{topo}");
44    /// }
45    /// ```
46    pub fn topology(&self, mpi: &crate::Mpi) -> Result<crate::TopologyInfo> {
47        crate::topology::gather_topology(self, mpi)
48    }

50    /// Get the group associated with this communicator.
51    ///
52    /// Returns a [`Group`] containing all processes in this communicator.
53    /// Use [`Group::include`] or [`Group::exclude`] on the returned group
54    /// to derive sub-groups, which can then be used with
55    /// `MPI_Comm_create_group` (a future epic).
56    ///
57    /// # Errors
58    ///
59    /// Returns an error if the underlying `MPI_Comm_group` call fails or if
60    /// the C-side group table is full.
61    ///
62    /// # Example
63    ///
64    /// ```no_run
65    /// use ferrompi::Mpi;
66    ///
67    /// let mpi = Mpi::init().unwrap();
68    /// let world = mpi.world();
69    /// let g = world.group().unwrap();
70    /// assert_eq!(g.size().unwrap(), world.size());
71    /// ```
72    pub fn group(&self) -> Result<Group> {
73        let mut group_handle: i32 = 0;
74        // SAFETY: self.handle is owned by this Communicator; group_handle is an exclusive output.
75        let ret = unsafe { ffi::ferrompi_comm_group(self.handle, &mut group_handle) };
76        Error::check_with_op(ret, "comm_group")?;
77        Ok(Group {
78            handle: group_handle,
79        })
80    }

82    /// Duplicate this communicator.
83    pub fn duplicate(&self) -> Result<Self> {
84        let mut new_handle: i32 = 0;
85        // SAFETY: self.handle is owned by this Communicator; new_handle is an exclusive output.
86        let ret = unsafe { ffi::ferrompi_comm_dup(self.handle, &mut new_handle) };
87        Error::check_with_op(ret, "comm_dup")?;
88        Self::from_handle(new_handle)
89    }

91    /// Split this communicator into sub-communicators based on color and key.
92    ///
93    /// Processes with the same `color` are placed in the same new communicator.
94    /// The `key` controls the rank ordering within the new communicator.
95    ///
96    /// Returns `None` if this process used [`Communicator::UNDEFINED`] as color.
97    ///
98    /// # Example
99    ///
100    /// ```no_run
101    /// use ferrompi::Mpi;
102    ///
103    /// let mpi = Mpi::init().unwrap();
104    /// let world = mpi.world();
105    /// let color = world.rank() % 2; // Even/odd split
106    /// if let Some(sub) = world.split(color, world.rank()).unwrap() {
107    ///     println!("Rank {} in sub-communicator of size {}", sub.rank(), sub.size());
108    /// }
109    /// ```
110    pub fn split(&self, color: i32, key: i32) -> Result<Option<Communicator>> {
111        let mut new_handle: i32 = 0;
112        // SAFETY: self.handle is owned by this Communicator; remaining arguments are scalars.
113        let ret = unsafe { ffi::ferrompi_comm_split(self.handle, color, key, &mut new_handle) };
114        Error::check_with_op(ret, "comm_split")?;
115        if new_handle < 0 {
116            Ok(None)
117        } else {
118            Self::from_handle(new_handle).map(Some)
119        }
120    }

122    /// Split this communicator by type.
123    ///
124    /// Processes that share the same resource (determined by `split_type`) are
125    /// placed in the same new communicator. The `key` controls the rank ordering
126    /// within the new communicator.
127    ///
128    /// Returns `None` if MPI returns `MPI_COMM_NULL` for this process.
129    ///
130    /// # Example
131    ///
132    /// ```no_run
133    /// use ferrompi::{Mpi, SplitType};
134    ///
135    /// let mpi = Mpi::init().unwrap();
136    /// let world = mpi.world();
137    /// if let Some(node) = world.split_type(SplitType::Shared, world.rank()).unwrap() {
138    ///     println!("Node has {} processes", node.size());
139    /// }
140    /// ```
141    pub fn split_type(&self, split_type: SplitType, key: i32) -> Result<Option<Communicator>> {
142        let mut new_handle: i32 = 0;
143        // SAFETY: self.handle is owned by this Communicator; remaining arguments are scalars.
144        let ret = unsafe {
145            ffi::ferrompi_comm_split_type(self.handle, split_type as i32, key, &mut new_handle)
146        };
147        Error::check_with_op(ret, "comm_split_type")?;
148        if new_handle < 0 {
149            Ok(None)
150        } else {
151            Self::from_handle(new_handle).map(Some)
152        }
153    }

155    /// Create a communicator containing only processes that share memory.
156    ///
157    /// This is equivalent to `split_type(SplitType::Shared, self.rank())`.
158    /// All processes on the same physical node will be in the same communicator.
159    ///
160    /// # Errors
161    ///
162    /// Returns [`Error::Internal`] if MPI unexpectedly returns a null communicator,
163    /// which should not happen for `MPI_COMM_TYPE_SHARED` under normal conditions.
164    ///
165    /// # Example
166    ///
167    /// ```no_run
168    /// use ferrompi::Mpi;
169    ///
170    /// let mpi = Mpi::init().unwrap();
171    /// let world = mpi.world();
172    /// let node = world.split_shared().unwrap();
173    /// println!("Node has {} processes, I am local rank {}", node.size(), node.rank());
174    /// ```
175    pub fn split_shared(&self) -> Result<Communicator> {
176        self.split_type(SplitType::Shared, self.rank())?
177            .ok_or_else(|| Error::Internal("split_shared returned null communicator".into()))
178    }

180    /// Create a sub-communicator from a group.
181    ///
182    /// `group` must be a subset of `self`'s group. This call is collective
183    /// over `self` — every rank in the parent communicator must invoke it,
184    /// even ranks not in `group`.
185    ///
186    /// Returns `Ok(None)` for ranks that are not members of `group`
187    /// (MPI returns `MPI_COMM_NULL` for those ranks). Returns
188    /// `Ok(Some(comm))` for ranks that are members.
189    ///
190    /// # Errors
191    ///
192    /// Returns an error if `group` contains a rank not present in the parent
193    /// communicator (`MPI_ERR_GROUP`), or if the C-side communicator table is
194    /// full (`MPI_ERR_OTHER`).
195    ///
196    /// # Example
197    ///
198    /// ```no_run
199    /// use ferrompi::Mpi;
200    ///
201    /// let mpi = Mpi::init().unwrap();
202    /// let world = mpi.world();
203    /// let g = world.group().unwrap().include(&[0, 2]).unwrap();
204    /// // Collective: every rank calls create_from_group
205    /// if let Some(sub) = world.create_from_group(&g).unwrap() {
206    ///     assert_eq!(sub.size(), 2);
207    /// }
208    /// ```
209    pub fn create_from_group(&self, group: &Group) -> Result<Option<Communicator>> {
210        let mut new_handle: i32 = -1;
211        // SAFETY: self.handle is owned by this Communicator; group.handle is owned by the Group
212        // argument and remains valid for the duration of this call.
213        let ret = unsafe {
214            ffi::ferrompi_comm_create_from_group_parent(self.handle, group.handle, &mut new_handle)
215        };
216        Error::check_with_op(ret, "comm_create_from_group_parent")?;
217        if new_handle < 0 {
218            Ok(None)
219        } else {
220            Self::from_handle(new_handle).map(Some)
221        }
222    }

224    /// Abort MPI execution across all processes in this communicator.
225    ///
226    /// This function terminates all processes associated with the communicator.
227    /// It calls `MPI_Abort` and then exits the process. This function never
228    /// returns.
229    ///
230    /// # Arguments
231    ///
232    /// * `errorcode` - Error code to return to the invoking environment
233    ///
234    /// # Example
235    ///
236    /// ```no_run
237    /// # use ferrompi::Mpi;
238    /// # let mpi = Mpi::init().unwrap();
239    /// # let world = mpi.world();
240    /// if world.rank() == 0 {
241    ///     // Fatal error detected, abort all processes
242    ///     world.abort(1);
243    /// }
244    /// ```
245    ///
246    /// # Fallback Behavior
247    ///
248    /// In the standard case `MPI_Abort` terminates the entire process
249    /// group and this function never returns. If `MPI_Abort` itself
250    /// returns (non-standard but observed in some implementations during
251    /// internal failures), this function falls back to
252    /// [`std::process::abort`], which raises `SIGABRT` and terminates
253    /// the current process immediately. The original `errorcode` is lost
254    /// in that fallback path because `process::abort` does not accept an
255    /// exit code; the signal code (usually 134 = 128 + 6) is the best
256    /// evidence the caller has that the fallback triggered.
257    pub fn abort(&self, errorcode: i32) -> ! {
258        // SAFETY: ferrompi_abort delegates to MPI_Abort, which is
259        // defined to terminate all processes in the communicator and
260        // never return. This call is safe to make with any valid
261        // communicator handle; self.handle is always valid because
262        // Communicator is only constructed through safe methods that
263        // guarantee handle validity.
264        unsafe { ffi::ferrompi_abort(self.handle, errorcode) };
265
266        // Defense in depth: the MPI standard says MPI_Abort "should"
267        // terminate all processes but does not strictly guarantee the
268        // calling process aborts before return. If MPI_Abort ever
269        // returns (non-standard implementation behavior), we must still
270        // honor the `-> !` contract. std::process::abort() raises
271        // SIGABRT and is guaranteed to diverge without running
272        // destructors — which is the right outcome, because any
273        // destructor that touches MPI state after a failed MPI_Abort
274        // has undefined behavior.
275        std::process::abort()
276    }
