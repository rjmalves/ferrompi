28/// An MPI communicator.
29///
30/// This type wraps an MPI communicator handle and provides safe methods for
31/// collective and point-to-point communication operations.
32///
33/// # Thread Safety
34///
35/// `Communicator` is `Send + Sync`, matching the thread-safety model of
36/// C/Fortran MPI. The actual thread-safety guarantees depend on the
37/// thread level provided by [`Mpi::init_thread()`](crate::Mpi::init_thread):
38///
39/// - [`ThreadLevel::Single`](crate::ThreadLevel::Single) /
40///   [`Funneled`](crate::ThreadLevel::Funneled): MPI calls only from main thread
41/// - [`ThreadLevel::Serialized`](crate::ThreadLevel::Serialized): MPI calls
42///   from any thread, but serialized by the user (e.g., via a `Mutex`)
43/// - [`ThreadLevel::Multiple`](crate::ThreadLevel::Multiple): MPI calls from
44///   any thread concurrently without external synchronization
45///
46/// For hybrid MPI + threads programs, request at least
47/// [`ThreadLevel::Funneled`](crate::ThreadLevel::Funneled) (master thread
48/// makes MPI calls) or [`ThreadLevel::Serialized`](crate::ThreadLevel::Serialized)
49/// (any thread, but only one at a time).
50///
51/// # Example
52///
53/// ```no_run
54/// use ferrompi::Mpi;
55///
56/// let mpi = Mpi::init().unwrap();
57/// let world = mpi.world();
58///
59/// println!("I am rank {} of {}", world.rank(), world.size());
60/// ```
61pub struct Communicator {
62    pub(crate) handle: i32,
63    pub(crate) rank: i32,
64    pub(crate) size: i32,
65}
66
67// SAFETY: Communicator handles are integer indices into a C-side table.
68// The C MPI library manages its own thread safety based on the thread level
69// requested via MPI_Init_thread. Sending a Communicator to another thread is
70// safe because MPI_Comm operations are defined to be callable from any thread
71// when the appropriate thread level (Serialized or Multiple) was requested.
72// Users must ensure they requested sufficient thread support and serialize
73// access themselves when using ThreadLevel::Serialized.
74//
75// NOTE: The C-layer handle tables (comm_table, request_table, etc.) are not
76// internally synchronized. At ThreadLevel::Single and Funneled, only the main
77// thread calls MPI so no data race occurs. At Serialized, the user is required
78// to serialize all MPI calls. At Multiple, concurrent MPI calls are safe per
79// the MPI standard, but our handle table allocation/deallocation is not
80// protected by mutexes. This is acceptable because MPI_THREAD_MULTIPLE only
81// guarantees that MPI calls are thread-safe — the handle tables are modified
82// only inside MPI entry/exit points which the user serializes or the MPI
83// implementation serializes internally.
84unsafe impl Send for Communicator {}
85unsafe impl Sync for Communicator {}
86
87impl Communicator {
88    /// Constant for opting out of a communicator split.
89    ///
90    /// Processes passing this as the `color` to [`split()`](Self::split) will not be
91    /// included in any resulting communicator.
92    pub const UNDEFINED: i32 = -1;
93
94    /// Get a handle to `MPI_COMM_WORLD`.
95    pub(crate) fn world() -> Self {
96        // SAFETY: ferrompi_comm_world() returns the well-known COMM_WORLD handle.
97        // MPI must be initialized before this is called (enforced by Mpi::world()).
98        let handle = unsafe { ffi::ferrompi_comm_world() };
99        Self::from_handle(handle).expect("COMM_WORLD must be valid post-init")
100    }
101
102    /// Construct a `Communicator` by querying rank and size once from MPI.
103    ///
104    /// This is the canonical internal constructor. All construction sites use it
105    /// so that `rank` and `size` are cached at creation time and subsequent calls
106    /// to [`rank()`](Self::rank) / [`size()`](Self::size) require no FFI round-trip.
107    ///
108    /// Returns `Err` if either `MPI_Comm_rank` or `MPI_Comm_size` fails.
109    pub(crate) fn from_handle(handle: i32) -> Result<Self> {
110        let mut rank: i32 = 0;
111        // SAFETY: `handle` is a valid MPI communicator handle obtained from MPI
112        // functions. `rank` is a local variable so the pointer is valid for writes.
113        let ret = unsafe { ffi::ferrompi_comm_rank(handle, &mut rank) };
114        Error::check_with_op(ret, "comm_rank")?;
115        let mut size: i32 = 0;
116        // SAFETY: same as above — `size` is a local variable, `handle` is valid.
117        let ret = unsafe { ffi::ferrompi_comm_size(handle, &mut size) };
118        Error::check_with_op(ret, "comm_size")?;
119        Ok(Communicator { handle, rank, size })
120    }
121
122    /// Get the raw communicator handle (for advanced use).
123    pub fn raw_handle(&self) -> i32 {
124        self.handle
125    }
126
127    /// Get the rank of the calling process in this communicator.
128    ///
129    /// Returns the cached value stored at construction time. No FFI call is made.
130    #[inline]
131    pub fn rank(&self) -> i32 {
132        self.rank
133    }
134
135    /// Get the number of processes in this communicator.
136    ///
137    /// Returns the cached value stored at construction time. No FFI call is made.
138    #[inline]
139    pub fn size(&self) -> i32 {
140        self.size
141    }
142}
143
144impl Drop for Communicator {
145    fn drop(&mut self) {
146        // Don't free COMM_WORLD (handle 0)
147        if self.handle != 0 {
148            unsafe { ffi::ferrompi_comm_free(self.handle) };
149        }
150    }
151}

164mod tests {

167    /// Helper: create a Communicator with handle 0 (COMM_WORLD).
168    /// Drop for handle 0 is a no-op, so this is safe without MPI.
169    /// `size: 1` prevents divide-by-zero in tests that do `buf.len() / comm.size()`.
170    fn dummy_comm() -> Communicator {
171        Communicator {
172            handle: 0,
173            rank: 0,
174            size: 1,
175        }
176    }
177
178    #[test]
179    fn cached_rank_size_returns_field_values() {
180        // Construct directly with arbitrary rank/size values to prove that
181        // rank() and size() return the cached fields without any FFI call.
182        let comm = Communicator {
183            handle: 0,
184            rank: 7,
185            size: 42,
186        };
187        assert_eq!(comm.rank(), 7);
188        assert_eq!(comm.size(), 42);
189    }
190
191    #[test]
192    fn split_type_shared_repr_value_and_traits() {
193        // SplitType::Shared has repr value 0
194        assert_eq!(SplitType::Shared as i32, 0);
195
196        // Clone works (Copy implies Clone)
197        let st = SplitType::Shared;
198        let cloned = st;
199        assert_eq!(cloned, SplitType::Shared);
200
201        // Debug works
202        assert_eq!(format!("{:?}", st), "Shared");
203    }
204
205    #[test]
206    fn communicator_undefined_is_negative_one() {
207        assert_eq!(Communicator::UNDEFINED, -1);
208    }

212        let comm = dummy_comm();
