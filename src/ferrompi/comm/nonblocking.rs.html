15    /// Nonblocking broadcast.
16    ///
17    /// Returns a request handle that must be waited on before accessing the buffer.
18    ///
19    /// # Safety Note
20    ///
21    /// The buffer must remain valid until the request is completed.
22    ///
23    /// # Example
24    ///
25    /// ```no_run
26    /// # use ferrompi::Mpi;
27    /// # let mpi = Mpi::init().unwrap();
28    /// # let world = mpi.world();
29    /// let mut data = vec![0.0f64; 100];
30    /// let req = world.ibroadcast(&mut data, 0).unwrap();
31    /// // ... do other work ...
32    /// req.wait().unwrap();
33    /// ```
34    pub fn ibroadcast<T: MpiDatatype>(&self, data: &mut [T], root: i32) -> Result<Request> {
35        let mut request_handle: i64 = 0;
36        let ret = unsafe {
37            ffi::ferrompi_ibcast(
38                data.as_mut_ptr().cast::<std::ffi::c_void>(),
39                data.len() as i64,
40                T::TAG as i32,
41                root,
42                self.handle,
43                &mut request_handle,
44            )
45        };
46        Error::check_with_op(ret, "ibcast")?;
47        Ok(Request::new(request_handle))
48    }

379    ///
380    /// Initiates an all-to-all operation and returns immediately with a
381    /// [`Request`] handle.
382    ///
383    /// # Errors
384    ///
385    /// Returns [`Error::InvalidBuffer`] if `send.len() != recv.len()` or

388    /// # Example
389    ///
390    /// ```no_run

392    /// # let mpi = Mpi::init().unwrap();
393    /// # let world = mpi.world();

401        let size = self.size() as usize;
402        if send.len() != recv.len() || send.len() % size != 0 {
403            return Err(Error::InvalidBuffer);
404        }
405        let count = (send.len() / size) as i64;
406        let mut request_handle: i64 = 0;
407        let ret = unsafe {
408            ffi::ferrompi_ialltoall(
409                send.as_ptr().cast::<std::ffi::c_void>(),
410                count,
411                recv.as_mut_ptr().cast::<std::ffi::c_void>(),
412                count,
413                T::TAG as i32,
414                self.handle,
415                &mut request_handle,
416            )
417        };
418        Error::check_with_op(ret, "ialltoall")?;
419        Ok(Request::new(request_handle))

471    /// Nonblocking in-place gather at root. Non-root ranks must use
472    /// `igather` — this method returns `Error::InvalidOp` on non-root.
473    ///
474    /// # Buffer Layout (root)
475    ///
476    /// `data` must have length `recvcount * size()` where `recvcount` is
477    /// the per-rank count. Rank `r`'s contribution lives at offset
478    /// `r * recvcount`. Root's own contribution must be pre-written into
479    /// `data[rank() * recvcount .. (rank()+1) * recvcount]` before the call.
480    ///
481    /// # Errors
482    ///
483    /// - `Error::InvalidOp` if this rank is not `root`.
484    /// - `Error::InvalidBuffer` if `data.len()` is not divisible by `size()`.
485    ///
486    /// # Example
487    ///
488    /// ```no_run
489    /// # use ferrompi::Mpi;
490    /// # let mpi = Mpi::init().unwrap();
491    /// # let world = mpi.world();
492    /// if world.rank() == 0 {
493    ///     let mut data = vec![0i32; 4 * world.size() as usize];
494    ///     let req = world.igather_inplace(&mut data, 0).unwrap();
495    ///     req.wait().unwrap();
496    /// }
497    /// ```
498    pub fn igather_inplace<T: MpiDatatype>(&self, data: &mut [T], root: i32) -> Result<Request> {
499        if self.rank() != root {
500            return Err(Error::InvalidOp);
501        }
502        let size = self.size() as usize;
503        if size == 0 || data.len() % size != 0 {
504            return Err(Error::InvalidBuffer);
505        }
506        let recvcount = (data.len() / size) as i64;
507        let mut request_handle: i64 = 0;
508        let ret = unsafe {
509            // SAFETY: data is a valid, exclusively-owned mutable slice. We cast to *mut c_void
510            // as required by the C FFI. The guard above guarantees self.rank() == root, so
511            // is_root is hardcoded to 1. The buffer outlives the returned Request handle;
512            // the caller must call wait() before accessing or dropping the buffer.
513            ffi::ferrompi_igather_inplace(
514                data.as_mut_ptr().cast::<std::ffi::c_void>(),
515                recvcount,
516                T::TAG as i32,
517                root,
518                1, // is_root = true by the rank guard above
519                self.handle,
520                &mut request_handle,
521            )
522        };
523        Error::check_with_op(ret, "igather_inplace")?;
524        Ok(Request::new(request_handle))
525    }

576    /// Nonblocking in-place scatter. At root, `data` is the `sendcount * size()`
577    /// send buffer; root's own slot is retained in place. At non-root, `data` is
578    /// the `recvcount`-element receive buffer.
579    ///
580    /// # Buffer Layout (root)
581    ///
582    /// `data` must have length `sendcount * size()`. Rank `r`'s slot is
583    /// `data[r*sendcount .. (r+1)*sendcount]`. After the wait, only root's own
584    /// slot is guaranteed to remain intact.
585    ///
586    /// # Errors
587    ///
588    /// - `Error::InvalidBuffer` at root if `data.len()` is not divisible by `size()`.
589    ///
590    /// # Example
591    ///
592    /// ```no_run
593    /// # use ferrompi::Mpi;
594    /// # let mpi = Mpi::init().unwrap();
595    /// # let world = mpi.world();
596    /// if world.rank() == 0 {
597    ///     let mut data = vec![0i32, 10, 20, 30];
598    ///     let req = world.iscatter_inplace(&mut data, 0).unwrap();
599    ///     req.wait().unwrap();
600    /// } else {
601    ///     let mut data = vec![0i32; 1];
602    ///     let req = world.iscatter_inplace(&mut data, 0).unwrap();
603    ///     req.wait().unwrap();
604    /// }
605    /// ```
606    pub fn iscatter_inplace<T: MpiDatatype>(&self, data: &mut [T], root: i32) -> Result<Request> {
607        let is_root = self.rank() == root;
608        let size = self.size() as usize;
609        let (sendbuf, sendcount, recvbuf, recvcount, is_root_flag) = if is_root {
610            if size == 0 || data.len() % size != 0 {
611                return Err(Error::InvalidBuffer);
612            }
613            let per = (data.len() / size) as i64;
614            (
615                data.as_ptr().cast::<std::ffi::c_void>(),
616                per,
617                std::ptr::null_mut::<std::ffi::c_void>(),
618                0i64,
619                1i32,
620            )
621        } else {
622            (
623                std::ptr::null::<std::ffi::c_void>(),
624                0i64,
625                data.as_mut_ptr().cast::<std::ffi::c_void>(),
626                data.len() as i64,
627                0i32,
628            )
629        };
630        let mut request_handle: i64 = 0;
631        let ret = unsafe {
632            // SAFETY: At root, sendbuf points to valid data of length sendcount*size elements
633            // (guaranteed by the divisibility check above); recvbuf is null (MPI_IN_PLACE path).
634            // At non-root, recvbuf points to a valid mutable slice of length recvcount elements;
635            // sendbuf is null (MPI standard ignores sendbuf on non-root scatter). Both pointers
636            // are cast to *const/*mut c_void as required by the C FFI. The buffer outlives the
637            // returned Request; the caller must call wait() before accessing or dropping it.
638            ffi::ferrompi_iscatter_inplace(
639                sendbuf,
640                sendcount,
641                recvbuf,
642                recvcount,
643                T::TAG as i32,
644                root,
645                is_root_flag,
646                self.handle,
647                &mut request_handle,
648            )
649        };
650        Error::check_with_op(ret, "iscatter_inplace")?;
651        Ok(Request::new(request_handle))
652    }
653
654    /// Nonblocking in-place all-to-all personalized communication. `data` is
655    /// both send and receive buffer on every rank.
656    ///
657    /// Before the call, rank `r` must pre-write into slot `s` (at offset
658    /// `s * count`) the payload it wishes to send to rank `s`. After the wait,
659    /// the same slot contains the data received FROM rank `s`.
660    ///
661    /// # Buffer Layout
662    ///
663    /// `data` must have length `count * size()`. Slot `s` at
664    /// `data[s*count..(s+1)*count]` holds data sent to (and later received from)
665    /// rank `s`.
666    ///
667    /// # Errors
668    ///
669    /// - `Error::InvalidBuffer` if `data.len()` is not divisible by `size()`.
670    ///
671    /// # Example
672    ///
673    /// ```no_run
674    /// # use ferrompi::Mpi;
675    /// # let mpi = Mpi::init().unwrap();
676    /// # let world = mpi.world();
677    /// let r = world.rank() as i32;
678    /// let size = world.size() as usize;
679    /// let mut data: Vec<i32> = (0..size as i32).map(|s| r * 10 + s).collect();
680    /// let req = world.ialltoall_inplace(&mut data).unwrap();
681    /// req.wait().unwrap();
682    /// ```
683    pub fn ialltoall_inplace<T: MpiDatatype>(&self, data: &mut [T]) -> Result<Request> {
684        let size = self.size() as usize;
685        if size == 0 || data.len() % size != 0 {
686            return Err(Error::InvalidBuffer);
687        }
688        let recvcount = (data.len() / size) as i64;
689        let mut request_handle: i64 = 0;
690        let ret = unsafe {
691            // SAFETY: data is a valid, exclusively-owned mutable slice of length recvcount*size
692            // elements (guaranteed by the divisibility check above). We cast to *mut c_void as
693            // required by the C FFI. MPI_IN_PLACE is passed as sendbuf in the C wrapper; the
694            // caller must pre-write each slot before calling this method. The buffer outlives
695            // the returned Request; the caller must call wait() before accessing or dropping it.
696            ffi::ferrompi_ialltoall_inplace(
697                data.as_mut_ptr().cast::<std::ffi::c_void>(),
698                recvcount,
699                T::TAG as i32,
700                self.handle,
701                &mut request_handle,
702            )
703        };
704        Error::check_with_op(ret, "ialltoall_inplace")?;
705        Ok(Request::new(request_handle))
706    }
707}
