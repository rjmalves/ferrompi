1//! # ferrompi
2//!
3//! Safe, generic Rust bindings for MPI (Message Passing Interface).
4//!
5//! This crate wraps MPI functionality through a thin C layer, providing:
6//! - Type-safe generic API for all MPI datatypes
7//! - Blocking, nonblocking, and persistent (MPI 4.0+) collectives
8//! - Communicator management (split, duplicate)
9//! - RMA shared memory windows (with `rma` feature)
10//! - SLURM environment helpers (with `numa` feature)
11//! - Large count support (MPI 4.0+ `_c` variants for blocking/nonblocking
12//!   collectives; persistent collectives currently reject `count > INT_MAX`
13//!   with `MPI_ERR_COUNT` — full `_c` dispatch for persistent ops is deferred)
14//!
15//! ## Supported Types
16//!
17//! All communication operations are generic over [`MpiDatatype`]:
18//! `f32`, `f64`, `i32`, `i64`, `u8`, `u32`, `u64`
19//!
20//! ## Quick Start
21//!
22//! ```no_run
23//! use ferrompi::{Mpi, ReduceOp};
24//!
25//! fn main() -> Result<(), ferrompi::Error> {
26//!     let mpi = Mpi::init()?;
27//!     let world = mpi.world();
28//!
29//!     let rank = world.rank();
30//!     let size = world.size();
31//!     println!("Hello from rank {} of {}", rank, size);
32//!
33//!     // Generic broadcast — works with any MpiDatatype
34//!     let mut data = vec![0.0f64; 100];
35//!     if rank == 0 {
36//!         data.fill(42.0);
37//!     }
38//!     world.broadcast(&mut data, 0)?;
39//!
40//!     // Generic all-reduce
41//!     let sum = world.allreduce_scalar(rank as f64, ReduceOp::Sum)?;
42//!     println!("Rank {rank}: sum of all ranks = {sum}");
43//!
44//!     Ok(())
45//! }
46//! ```
47//!
48//! ## Feature Flags
49//!
50//! | Feature | Description | Dependencies |
51//! |---------|-------------|--------------|
52//! | `rma`   | RMA shared memory window operations | — |
53//! | `numa`  | NUMA-aware windows and SLURM helpers | `rma` |
54//!
55//! ## Capabilities
56//!
57//! - **Generic API**: All operations work with any [`MpiDatatype`] (`f32`, `f64`, `i32`, `i64`, `u8`, `u32`, `u64`)
58//! - **Blocking collectives**: barrier, broadcast, reduce, allreduce, gather, scatter, allgather,
59//!   alltoall, scan, exscan, reduce\_scatter\_block, plus V-variants (gatherv, scatterv, allgatherv, alltoallv)
60//! - **Nonblocking collectives**: All 15 `i`-prefixed variants with [`Request`] handles
61//! - **Persistent collectives** (MPI 4.0+): All 15 `_init` variants with [`PersistentRequest`] handles
62//! - **Scalar and in-place variants**: `reduce_scalar`, `allreduce_scalar`, `reduce_inplace`,
63//!   `allreduce_inplace`, `scan_scalar`, `exscan_scalar`
64//! - **Point-to-point**: `send`, `recv`, `isend`, `irecv`, `sendrecv`, `probe`, `iprobe`
65//! - **Communicator management**: `split`, `split_type`, `split_shared`, `duplicate`
66//! - **Group operations**: [`Group`] with incl/excl/union/intersection/difference,
67//!   [`RankRange`] for range constructors, [`GroupComparison`].
68//!
69//!   Note: [`Mpi::create_from_group`] requires MPI 4.0+.
70//!   Support is probed once and cached; see the function rustdoc for the cache invariant.
71//! - **Custom datatypes**: [`CustomDatatype`]
72//!   (contiguous/vector/struct/resized) and [`StructField`]
73//!   for struct-type builders.
74//! - **User-defined reduction operations**: [`UserOp`] wraps `MPI_Op_create`
75//!   with safe closure storage and trampoline.
76//! - **Distributed RMA windows** (feature `rma`): [`Win<T>`](crate::Win) with
77//!   [`WinFenceAssert`], [`WinPscwAssert`],
78//!   [`WinLockGuard`], and [`WinLockAllGuard`]
79//!   RAII guards.
80//! - **Info objects**: [`Info`] for runtime hint passing to communicator,
81//!   window, and operation constructors.
82//! - **Persistent point-to-point**: `send_init`, `bsend_init`, `rsend_init`, `ssend_init`,
83//!   `recv_init` methods on [`Communicator`], each returning a
84//!   [`PersistentRequest`].
85//! - **Shared memory windows** (feature `rma`): [`SharedWindow<T>`](crate::SharedWindow)
86//!   with RAII lock guards for NUMA-aware intra-node shared memory (distinct from the
87//!   distributed [`Win<T>`](crate::Win) windows above).
88//! - **SLURM helpers** (feature `numa`): Job topology queries via `slurm` module
89//! - **Rich error handling**: [`MpiErrorClass`] categorization with messages from the MPI runtime
90//!
91//! ## Thread Safety
92//!
93//! [`Communicator`] is `Send + Sync` to support hybrid MPI + threads programs
94//! (e.g., MPI between nodes, `std::thread::scope` within a node).
95//!
96//! The actual thread-safety guarantees depend on the thread level requested
97//! at initialization:
98//!
99//! | Thread Level | Who can call MPI | Synchronization |
100//! |--------------|------------------|-----------------|
101//! | [`ThreadLevel::Single`] | Main thread only | N/A |
102//! | [`ThreadLevel::Funneled`] | Main thread only | N/A |
103//! | [`ThreadLevel::Serialized`] | Any thread | User must serialize |
104//! | [`ThreadLevel::Multiple`] | Any thread | None needed |
105//!
106//! ```no_run
107//! use ferrompi::{Mpi, ThreadLevel};
108//!
109//! // Request serialized thread support for hybrid MPI + threads
110//! let mpi = Mpi::init_thread(ThreadLevel::Funneled).unwrap();
111//! assert!(mpi.thread_level() >= ThreadLevel::Funneled);
112//! ```
113//!
114//! [`Mpi`] itself is `!Send + !Sync` — MPI initialization and finalization
115//! must occur on the same thread. Only [`Communicator`] handles (and the
116//! operations on them) may cross thread boundaries.
117//!
118//! ### Send/Sync Status of Public Types
119//!
120//! | Type | Send/Sync | Notes |
121//! |------|-----------|-------|
122//! | [`Communicator`] | `Send + Sync` | Explicit `unsafe impl` in `src/comm/mod.rs`; cross-thread use is the primary hybrid MPI use case. |
123//! | [`Mpi`] | `!Send + !Sync` | `PhantomData<*const ()>` field; init and finalize must occur on the same thread. |
124//! | [`Group`] | `Send + Sync` | Explicit `unsafe impl` in `src/group.rs`; handles are opaque integers, MPI-thread-safe under `MPI_THREAD_MULTIPLE`. |
125//! | [`Request`] | `Send + Sync` | Auto-derived; `i64` + `bool` fields. Cross-thread use requires `MPI_THREAD_MULTIPLE`. Buffer-lifetime invariant still applies. |
126//! | [`PersistentRequest`] | `Send + Sync` | Auto-derived; same shape as `Request`. ADR-0004 §"Drop behavior" applies across thread boundaries. |
127//! | [`Status`] | `Send + Sync` | POD wrapper; all fields are `Copy`. |
128//! | [`CustomDatatype`] | `Send + Sync` | Explicit `unsafe impl` in `src/datatype_builder.rs`; handle is an opaque integer. |
129//! | [`Info`] | `Send + Sync` | Auto-derived from `i32` + `bool` fields; MPI info objects are thread-safe under `MPI_THREAD_MULTIPLE`. |
130//! | [`UserOp<T>`] | `Send + Sync` (for `T: MpiDatatype`) | Auto-derived: fields are `i32` + `PhantomData<T>`. The trait bound `MpiDatatype: Copy + Send + 'static` and the fact that all concrete `MpiDatatype` impls are also `Sync` give `Send + Sync` for `UserOp<T>`. The global closure registry uses internal `unsafe impl Send/Sync` on its slots; that is a separate object from `UserOp<T>` itself. |
131//! | [`Win<T>`](crate::Win) (feature `rma`) | `!Send + !Sync` | `NonNull<T>` field suppresses auto-traits; RMA window's local memory pointer is not safe to share across threads. |
132//! | [`SharedWindow<T>`](crate::SharedWindow) (feature `rma`) | `!Send + !Sync` | `NonNull<T>` field; same rationale as `Win<T>`. |
133//! | [`LockGuard<'a, T>`] (feature `rma`) | `!Send + !Sync` | Borrows `SharedWindow<T>`; inherits non-Send/Sync. |
134//! | [`LockAllGuard<'a, T>`] (feature `rma`) | `!Send + !Sync` | Borrows `SharedWindow<T>`; inherits non-Send/Sync. |
135//! | [`WinLockGuard<'g, 'a, T>`](crate::WinLockGuard) (feature `rma`) | `!Send + !Sync` | Borrows `Win<T>`; inherits non-Send/Sync. |
136//! | [`WinLockAllGuard<'g, 'a, T>`](crate::WinLockAllGuard) (feature `rma`) | `!Send + !Sync` | Borrows `Win<T>`; inherits non-Send/Sync. |
137//!
138//! ## Hybrid MPI+OpenMP
139//!
140//! For hybrid parallelism, use [`Mpi::init_thread()`] with the appropriate level:
141//!
142//! - **[`Funneled`](ThreadLevel::Funneled)** (recommended): Only the main thread makes MPI calls.
143//!   OpenMP threads handle computation between MPI calls.
144//! - **[`Serialized`](ThreadLevel::Serialized)**: Any thread can make MPI calls, but only one at a time.
145//! - **[`Multiple`](ThreadLevel::Multiple)**: Full concurrent MPI from any thread (highest overhead).
146//!
147//! ```no_run
148//! use ferrompi::{Mpi, ThreadLevel, ReduceOp};
149//!
150//! let mpi = Mpi::init_thread(ThreadLevel::Funneled).unwrap();
151//! assert!(mpi.thread_level() >= ThreadLevel::Funneled);
152//!
153//! let world = mpi.world();
154//! // Worker threads compute locally, main thread calls MPI
155//! let local = 42.0_f64;
156//! let global = world.allreduce_scalar(local, ReduceOp::Sum).unwrap();
157//! ```
158//!
159//! ### SLURM Configuration
160//!
161//! ```bash
162//! #SBATCH --ntasks-per-node=4        # MPI ranks per node
163//! #SBATCH --cpus-per-task=8          # OpenMP threads per rank
164//! #SBATCH --bind-to core             # Pin MPI ranks
165//! export OMP_NUM_THREADS=$SLURM_CPUS_PER_TASK
166//! srun ./my_program
167//! ```
168//!
169//! Use the `slurm` module (with `numa` feature) to read these values at runtime.
170//! See `examples/hybrid_openmp.rs` for the full pattern.

349/// MPI environment handle.
350///
351/// This type represents an initialized MPI environment. There can only be one
352/// instance of this type at a time. When dropped, it finalizes MPI.
353///
354/// # Example
355///
356/// ```no_run
357/// use ferrompi::Mpi;
358///
359/// let mpi = Mpi::init().expect("Failed to initialize MPI");
360/// let world = mpi.world();
361/// println!("Running on {} processes", world.size());
362/// // MPI is finalized when `mpi` goes out of scope
363/// ```
364pub struct Mpi {
365    /// The thread level that was provided
366    thread_level: ThreadLevel,
367    /// Marker to make Mpi !Send and !Sync
368    _marker: PhantomData<*const ()>,
369}

371impl Mpi {
372    /// Initialize MPI with single-threaded support.
373    ///
374    /// # Errors
375    ///
376    /// Returns an error if MPI is already initialized or if initialization fails.
377    pub fn init() -> Result<Self> {
378        Self::init_thread(ThreadLevel::Single)
379    }
380
381    /// Initialize MPI with the specified thread support level.
382    ///
383    /// # Arguments
384    ///
385    /// * `required` - The minimum thread support level required
386    ///
387    /// # Returns
388    ///
389    /// Returns the MPI handle. The actual thread support level provided can be
390    /// queried with [`thread_level()`](Self::thread_level).
391    ///
392    /// # Errors
393    ///
394    /// Returns an error if MPI is already initialized or if initialization fails.
395    pub fn init_thread(required: ThreadLevel) -> Result<Self> {
396        // Check if already initialized
397        if MPI_INITIALIZED.swap(true, Ordering::SeqCst) {
398            return Err(Error::AlreadyInitialized);
399        }
400
401        let mut provided: i32 = 0;
402        let ret = unsafe { ffi::ferrompi_init_thread(required as i32, &mut provided) };
403
404        if ret != 0 {
405            MPI_INITIALIZED.store(false, Ordering::SeqCst);
406            return Err(Error::Mpi {
407                class: MpiErrorClass::Raw(ret),
408                code: ret,
409                message: format!("MPI_Init_thread failed with code {ret}"),
410                operation: Some("init_thread"),
411            });
412        }
413
414        let thread_level = match provided {
415            0 => ThreadLevel::Single,
416            1 => ThreadLevel::Funneled,
417            2 => ThreadLevel::Serialized,
418            _ => ThreadLevel::Multiple,
419        };
420
421        Ok(Mpi {
422            thread_level,
423            _marker: PhantomData,
424        })
425    }

444    /// Get the MPI library version string (implementation-specific).
445    ///
446    /// Returns a string such as `"Open MPI v4.1.6"` or `"Intel(R) MPI Library 2021.7"`.
447    /// This wraps `MPI_Get_library_version`.
448    pub fn library_version() -> Result<String> {
449        // MPI_MAX_LIBRARY_VERSION_STRING is 8192 in most implementations.
450        let mut buf = [0u8; 8192];
451        let mut len: i32 = 0;
452        let ret = unsafe {
453            ffi::ferrompi_get_library_version(buf.as_mut_ptr().cast::<c_char>(), &mut len)
454        };
455        Error::check_with_op(ret, "get_library_version")?;
456        let len = (len.max(0) as usize).min(buf.len());
457        // Trim trailing whitespace/newlines that some implementations append.
458        let s = std::str::from_utf8(&buf[..len])
459            .map_err(|_| Error::Internal("Invalid UTF-8 in library version string".into()))?;
460        Ok(s.trim_end().to_string())
461    }

463    /// Get the MPI standard version string (e.g., "MPI 4.0").
464    pub fn version() -> Result<String> {
465        let mut buf = [0u8; 256];
466        let mut len: i32 = 0;
467        let ret = unsafe { ffi::ferrompi_get_version(buf.as_mut_ptr().cast::<c_char>(), &mut len) };
468
469        if ret != 0 {
470            return Err(Error::from_code_with_op(ret, "get_version"));
471        }
472
473        let len = (len.max(0) as usize).min(buf.len());
474        let s = std::str::from_utf8(&buf[..len])
475            .map_err(|_| Error::Internal("Invalid UTF-8 in version string".into()))?;
476        Ok(s.to_string())
477    }

493    /// Returns `true` if the runtime MPI version is 4.0 or later.
494    ///
495    /// Caching semantics:
496    /// - An `Err` from `Mpi::version()` (e.g., called before `Mpi::init`)
497    ///   is NOT cached; a subsequent call after init can re-probe and
498    ///   observe support correctly.
499    /// - A successful `version()` whose string parses to a major version
500    ///   `≥ 4` caches `true`.
501    /// - A successful `version()` whose string parses to a major version
502    ///   `< 4` (including unrecognized formats that yield major `= 0` via
503    ///   the `unwrap_or(0)` fallback) caches `false`. Re-probing is not
504    ///   possible once a successful `version()` has been seen, so a
505    ///   non-standard version string format from an unusual MPI build
506    ///   permanently disables `Mpi::create_from_group` for this process.
507    fn supports_create_from_group() -> bool {
508        static SUPPORTED: OnceLock<bool> = OnceLock::new();
509        if let Some(&cached) = SUPPORTED.get() {
510            return cached;
511        }
512        // Probe. If `version()` fails, return `false` without caching;
513        // a future call can re-probe successfully.
514        let Ok(v) = Mpi::version() else {
515            return false;
516        };
517        // Mpi::version() returns a string like "MPI 4.0" or "MPI 3.1".
518        // Extract the major version number from the second whitespace-delimited
519        // token, then its first dot-delimited component.
520        let major: u32 = v
521            .split_whitespace()
522            .nth(1)
523            .and_then(|tok| tok.split('.').next())
524            .and_then(|s| s.parse().ok())
525            .unwrap_or(0);
526        let supported = major >= 4;
527        // Race: if another thread already won set(), discard our value;
528        // both threads agree on the result anyway because the probe is
529        // deterministic given a successful version() call.
530        let _ = SUPPORTED.set(supported);
531        supported
532    }

536    ///
537    /// `stringtag` must be identical across all ranks that participate
538    /// in the call; ranks with different tags or in different groups
539    /// produce separate communicators.
540    ///
541    /// # Errors
542    ///
543    /// - Returns `Err(Error::Internal(_))` if `stringtag` contains a null byte
544    ///   (the FFI call is never invoked in this case).
545    /// - Returns `Err(Error::NotSupported("MPI_Comm_create_from_group"))` on
546    ///   MPI < 4.0 installations.
547    /// - Returns `Err(Error::Mpi { .. })` if the underlying MPI call fails.
548    ///
549    /// # Example
550    ///
551    /// ```no_run
552    /// use ferrompi::Mpi;
553    ///
554    /// let mpi = Mpi::init().unwrap();
555    /// let world = mpi.world();
556    /// let g = world.group().unwrap();
557    /// // All ranks participate with the same tag.
558    /// let comm = mpi.create_from_group(&g, "my-tag").unwrap();
559    /// assert_eq!(comm.size(), world.size());
560    /// ```
561    pub fn create_from_group(&self, group: &group::Group, stringtag: &str) -> Result<Communicator> {
562        let c_tag = CString::new(stringtag)
563            .map_err(|_| Error::Internal("stringtag contains null byte".into()))?;
564        if !Self::supports_create_from_group() {
565            return Err(Error::NotSupported(
566                "MPI_Comm_create_from_group".to_string(),
567            ));
568        }
569        let mut new_handle: i32 = -1;
570        // SAFETY: c_tag.as_ptr() is a valid, null-terminated C string that
571        // lives for the duration of this call. group.handle is a valid group
572        // handle obtained from ferrompi_comm_group or a group-constructor shim.
573        // &mut new_handle is a pointer to a stack-allocated i32.
574        let ret = unsafe {
575            ffi::ferrompi_comm_create_from_group(group.handle, c_tag.as_ptr(), &mut new_handle)
576        };
577        Error::check_with_op(ret, "comm_create_from_group")?;
578        Communicator::from_handle(new_handle)
579    }

581    /// Attach a user-provided buffer for use by buffered sends.
582    ///
583    /// This wraps `MPI_Buffer_attach`. Once attached, the buffer is owned by
584    /// MPI until [`buffer_detach`](Self::buffer_detach) is called. Only one
585    /// buffer may be attached per process at a time; attempting to attach a
586    /// second buffer without first detaching returns
587    /// `Err(`[`Error::InvalidOp`]`)`.
588    ///
589    /// The `buffer` is stored in a process-wide static so its allocation
590    /// remains valid for the lifetime of the attachment. You must not access
591    /// the raw bytes of `buffer` between `buffer_attach` and `buffer_detach` —
592    /// MPI owns the contents during that window.
593    ///
594    /// # Buffer Sizing
595    ///
596    /// The recommended buffer size for `N` buffered sends of `count` elements
597    /// of type `T` is:
598    ///
599    /// ```text
600    /// N * (MPI_BSEND_OVERHEAD + count * size_of::<T>())
601    /// ```
602    ///
603    /// `MPI_BSEND_OVERHEAD` is implementation-specific; use at least a few
604    /// hundred extra bytes per buffered send. For safety, use a generous margin.
605    ///
606    /// Buffers larger than `i32::MAX` bytes are rejected with
607    /// `Err(`[`Error::InvalidBuffer`]`)` before the FFI call; the
608    /// underlying `MPI_Buffer_attach` takes an `int` count and cannot
609    /// address larger buffers.
610    ///
611    /// # Errors
612    ///
613    /// - [`Error::InvalidBuffer`] if `buffer.len() > i32::MAX as usize`.
614    /// - [`Error::InvalidOp`] if a buffer is already attached.
615    /// - [`Error::Mpi`] if `MPI_Buffer_attach` fails.
616    ///
617    /// # Example
618    ///
619    /// ```no_run
620    /// # use ferrompi::Mpi;
621    /// let mpi = Mpi::init().unwrap();
622    /// mpi.buffer_attach(vec![0u8; 64 * 1024].into_boxed_slice()).unwrap();
623    /// // ... buffered sends here ...
624    /// let _ = mpi.buffer_detach().unwrap();
625    /// ```
626    pub fn buffer_attach(&self, buffer: Box<[u8]>) -> Result<()> {
627        let mut guard = ATTACHED_BUFFER
628            .lock()
629            .map_err(|_| Error::Internal("ATTACHED_BUFFER mutex poisoned".into()))?;
630        if guard.is_some() {
631            return Err(Error::InvalidOp);
632        }
633        if buffer.len() > i32::MAX as usize {
634            return Err(Error::InvalidBuffer);
635        }
636        let ptr = buffer.as_ptr() as *mut std::ffi::c_void;
637        let size = buffer.len() as i64;
638        // Store the box in the static BEFORE calling MPI so the memory is
639        // guaranteed alive when MPI begins using the buffer.
640        *guard = Some(buffer);
641        // SAFETY: ptr points to the boxed slice we just stored in the static;
642        // the allocation remains valid until buffer_detach drops it. size is
643        // the exact byte length of that allocation.
644        let ret = unsafe { ffi::ferrompi_buffer_attach(ptr, size) };
645        if ret != 0 {
646            // Roll back: reclaim the box so the caller can retry.
647            guard.take();
648            return Err(Error::from_code_with_op(ret, "buffer_attach"));
649        }
650        Ok(())
651    }
652
653    /// Detach the previously attached buffer and return it to the caller.
654    ///
655    /// This wraps `MPI_Buffer_detach`. The call **blocks** until all buffered
656    /// sends that are currently using the buffer have completed. Once this
657    /// returns, the returned `Box<[u8]>` is owned by the caller again and
658    /// may be dropped or reused.
659    ///
660    /// Do **not** call this from a `Drop` implementation (e.g., on a wrapper
661    /// around `Mpi`). `MPI_Buffer_detach` blocks until pending sends drain;
662    /// blocking inside `Drop` can produce hard-to-diagnose hangs.
663    ///
664    /// # Errors
665    ///
666    /// - [`Error::InvalidOp`] if no buffer is currently attached.
667    /// - [`Error::Mpi`] if `MPI_Buffer_detach` fails.
668    ///
669    /// # Example
670    ///
671    /// ```no_run
672    /// # use ferrompi::Mpi;
673    /// let mpi = Mpi::init().unwrap();
674    /// mpi.buffer_attach(vec![0u8; 64 * 1024].into_boxed_slice()).unwrap();
675    /// // ... buffered sends ...
676    /// let _buf = mpi.buffer_detach().unwrap(); // dropped here
677    /// ```
678    pub fn buffer_detach(&self) -> Result<Box<[u8]>> {
679        let mut guard = ATTACHED_BUFFER
680            .lock()
681            .map_err(|_| Error::Internal("ATTACHED_BUFFER mutex poisoned".into()))?;
682        if guard.is_none() {
683            return Err(Error::InvalidOp);
684        }
685        let mut out_ptr: *mut std::ffi::c_void = std::ptr::null_mut();
686        let mut out_size: i64 = 0;
687        // SAFETY: out_ptr and out_size are valid stack-allocated output parameters.
688        // MPI_Buffer_detach writes the buffer pointer and its size into them.
689        let ret = unsafe { ffi::ferrompi_buffer_detach(&mut out_ptr, &mut out_size) };
690        if ret != 0 {
691            return Err(Error::from_code_with_op(ret, "buffer_detach"));
692        }
693        // Reclaim the Box from the static; this is the same allocation MPI just
694        // released. We take() here so the static is cleared atomically.
695        let buf = guard.take().expect("guard was Some; take() must succeed");
696        Ok(buf)
697    }
698}

700impl Drop for Mpi {
701    fn drop(&mut self) {
702        // Only finalize if we successfully initialized
703        if MPI_INITIALIZED.load(Ordering::SeqCst) {
704            unsafe {
705                ffi::ferrompi_finalize();
706            }
707            MPI_INITIALIZED.store(false, Ordering::SeqCst);
708        }
709    }
710}
