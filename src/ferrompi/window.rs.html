1//! Shared memory windows for intra-node communication.
2//!
3//! This module provides [`SharedWindow<T>`], a safe Rust wrapper around
4//! `MPI_Win_allocate_shared` with RAII lifecycle management. Shared memory
5//! windows allow processes on the same node to directly access each other's
6//! memory without explicit message passing, enabling high-performance
7//! intra-node communication.
8//!
9//! # Synchronization
10//!
11//! MPI shared memory windows require explicit synchronization:
12//!
13//! - **Active target**: Use [`SharedWindow::fence()`] for bulk-synchronous access
14//!   patterns where all processes participate in synchronization.
15//! - **Passive target**: Use [`SharedWindow::lock()`] or [`SharedWindow::lock_all()`]
16//!   for fine-grained, one-sided access patterns. These return RAII guards
17//!   ([`LockGuard`] / [`LockAllGuard`]) that automatically unlock on drop.
18//!
19//! # Feature Gate
20//!
21//! This module is only available when the `rma` feature is enabled.
22//!
23//! # Example
24//!
25//! ```no_run
26//! use ferrompi::{Mpi, SharedWindow, LockType};
27//!
28//! let mpi = Mpi::init().unwrap();
29//! let world = mpi.world();
30//! let node = world.split_shared().unwrap();
31//!
32//! // Each process allocates 100 f64s in shared memory
33//! let mut win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
34//!
35//! // Write to local portion
36//! {
37//!     let local = win.local_slice_mut();
38//!     for (i, x) in local.iter_mut().enumerate() {
39//!         *x = (node.rank() * 100 + i as i32) as f64;
40//!     }
41//! }
42//!
43//! // Fence synchronization
44//! win.fence().unwrap();
45//!
46//! // Read from any rank's memory
47//! let remote = win.remote_slice(0).unwrap();
48//! println!("Rank 0's first value: {}", remote[0]);
49//! ```

58/// Lock type for window synchronization.
59///
60/// Controls the level of concurrency allowed when locking an MPI window
61/// for passive target access.
62#[derive(Debug, Clone, Copy, PartialEq, Eq)]
63pub enum LockType {
64    /// Exclusive access (no other locks allowed).
65    ///
66    /// Only one process may hold an exclusive lock on a given rank's
67    /// window at a time. This is suitable for read-write access.
68    Exclusive,
69    /// Shared access (multiple shared locks allowed).
70    ///
71    /// Multiple processes may hold shared locks on a given rank's window
72    /// simultaneously. This is suitable for read-only access patterns.
73    Shared,
74}
75
76/// A shared memory window allocated across processes on the same node.
77///
78/// Created via [`SharedWindow::allocate()`] on a shared-memory communicator
79/// (typically obtained from [`Communicator::split_shared()`]).
80///
81/// Each process in the communicator contributes a local segment of `local_count`
82/// elements of type `T` to the shared memory region. All processes can then
83/// access any rank's segment via [`remote_slice()`](Self::remote_slice), provided
84/// proper synchronization is used.
85///
86/// # RAII Lifecycle
87///
88/// The underlying MPI window is freed automatically when the `SharedWindow`
89/// is dropped. The shared memory region becomes invalid after the window is freed.
90///
91/// # Thread Safety
92///
93/// `SharedWindow` is intentionally **not** `Send` or `Sync`. MPI windows have
94/// specific thread-safety rules that depend on the MPI thread level, and incorrect
95/// cross-thread usage can cause data corruption or MPI errors.
96///
97/// # Example
98///
99/// ```no_run
100/// use ferrompi::{Mpi, SharedWindow, LockType};
101///
102/// let mpi = Mpi::init().unwrap();
103/// let world = mpi.world();
104/// let node = world.split_shared().unwrap();
105///
106/// // Each process allocates 100 f64s in shared memory
107/// let mut win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
108///
109/// // Write to local portion
110/// {
111///     let local = win.local_slice_mut();
112///     for (i, x) in local.iter_mut().enumerate() {
113///         *x = (node.rank() * 100 + i as i32) as f64;
114///     }
115/// }
116///
117/// // Fence synchronization
118/// win.fence().unwrap();
119///
120/// // Read from any rank's memory
121/// let remote = win.remote_slice(0).unwrap();
122/// println!("Rank 0's first value: {}", remote[0]);
123/// ```
124pub struct SharedWindow<T: MpiDatatype> {
125    /// The MPI window handle (index into the C-layer window table).
126    win_handle: i32,
127    /// Non-null pointer to this process's local shared memory segment.
128    local_ptr: NonNull<T>,
129    /// Number of `T` elements in the local segment.
130    local_len: usize,
131    /// Number of processes in the communicator that created this window.
132    comm_size: i32,
133}
134
135impl<T: MpiDatatype> SharedWindow<T> {
136    /// Allocate a shared memory window.
137    ///
138    /// Each process in `comm` allocates `local_count` elements of type `T`
139    /// in a shared memory segment accessible by all processes in the communicator.
140    /// The communicator should be a shared-memory communicator (e.g., from
141    /// [`Communicator::split_shared()`]).
142    ///
143    /// # Arguments
144    ///
145    /// * `comm` - A shared-memory communicator (all processes must be on the same node)
146    /// * `local_count` - Number of elements of type `T` to allocate locally
147    ///
148    /// # Errors
149    ///
150    /// Returns an error if:
151    /// - The MPI window allocation fails (e.g., insufficient shared memory)
152    /// - The MPI implementation returns a null base pointer
153    ///
154    /// # Example
155    ///
156    /// ```no_run
157    /// use ferrompi::{Mpi, SharedWindow};
158    ///
159    /// let mpi = Mpi::init().unwrap();
160    /// let node = mpi.world().split_shared().unwrap();
161    /// let win = SharedWindow::<f64>::allocate(&node, 1024).unwrap();
162    /// ```
163    pub fn allocate(comm: &Communicator, local_count: usize) -> Result<Self> {
164        let size = (local_count * std::mem::size_of::<T>()) as i64;
165        let disp_unit = std::mem::size_of::<T>() as i32;
166        let mut baseptr: *mut std::ffi::c_void = std::ptr::null_mut();
167        let mut win_handle: i32 = 0;
168
169        // SAFETY: We pass valid pointers for out-parameters. The C layer
170        // allocates shared memory and returns a window handle + base pointer.
171        let ret = unsafe {
172            ffi::ferrompi_win_allocate_shared(
173                size,
174                disp_unit,
175                -1, // MPI_INFO_NULL
176                comm.raw_handle(),
177                &mut baseptr,
178                &mut win_handle,
179            )
180        };
181        Error::check(ret)?;
182
183        let local_ptr = NonNull::new(baseptr.cast::<T>())
184            .ok_or_else(|| Error::Internal("Win_allocate_shared returned null".into()))?;
185
186        Ok(SharedWindow {
187            win_handle,
188            local_ptr,
189            local_len: local_count,
190            comm_size: comm.size(),
191        })
192    }
193
194    /// Get a slice of this process's local shared memory segment.
195    ///
196    /// The returned slice provides read access to the `local_count` elements
197    /// that were allocated by this process in [`allocate()`](Self::allocate).
198    ///
199    /// # Safety Contract
200    ///
201    /// The caller must ensure proper MPI synchronization (fence or lock)
202    /// before reading data that may have been written by other processes.
203    pub fn local_slice(&self) -> &[T] {
204        // SAFETY: `local_ptr` was returned by MPI_Win_allocate_shared and is
205        // guaranteed valid for `local_len` elements of type T for the lifetime
206        // of the window. The pointer is non-null (checked in allocate()).
207        unsafe { std::slice::from_raw_parts(self.local_ptr.as_ptr(), self.local_len) }
208    }
209
210    /// Get a mutable slice of this process's local shared memory segment.
211    ///
212    /// The returned slice provides read-write access to the `local_count` elements
213    /// that were allocated by this process in [`allocate()`](Self::allocate).
214    ///
215    /// # Safety Contract
216    ///
217    /// The caller must ensure proper MPI synchronization (fence or lock)
218    /// before writing data that other processes may read.
219    pub fn local_slice_mut(&mut self) -> &mut [T] {
220        // SAFETY: `local_ptr` was returned by MPI_Win_allocate_shared and is
221        // guaranteed valid for `local_len` elements of type T for the lifetime
222        // of the window. We have `&mut self` ensuring exclusive access.
223        unsafe { std::slice::from_raw_parts_mut(self.local_ptr.as_ptr(), self.local_len) }
224    }
225
226    /// Query another rank's shared memory region.
227    ///
228    /// Returns a slice of the shared memory segment allocated by the specified
229    /// rank. This allows direct read access to remote memory without any
230    /// message passing overhead.
231    ///
232    /// # Arguments
233    ///
234    /// * `rank` - The rank whose shared memory to query (0-based)
235    ///
236    /// # Errors
237    ///
238    /// Returns an error if:
239    /// - The rank is out of range
240    /// - The MPI shared query call fails
241    ///
242    /// # Safety Contract
243    ///
244    /// The caller must ensure proper MPI synchronization (fence or lock)
245    /// before reading data that was written by the remote process.
246    ///
247    /// # Example
248    ///
249    /// ```no_run
250    /// # use ferrompi::{Mpi, SharedWindow};
251    /// # let mpi = Mpi::init().unwrap();
252    /// # let node = mpi.world().split_shared().unwrap();
253    /// # let win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
254    /// // Read rank 0's shared memory
255    /// let remote = win.remote_slice(0).unwrap();
256    /// println!("Rank 0 has {} elements, first = {}", remote.len(), remote[0]);
257    /// ```
258    pub fn remote_slice(&self, rank: i32) -> Result<&[T]> {
259        let mut size: i64 = 0;
260        let mut disp_unit: i32 = 0;
261        let mut baseptr: *mut std::ffi::c_void = std::ptr::null_mut();
262
263        // SAFETY: We pass valid pointers for out-parameters. The C layer queries
264        // the MPI window for the base pointer and size of the specified rank's
265        // shared memory segment.
266        let ret = unsafe {
267            ffi::ferrompi_win_shared_query(
268                self.win_handle,
269                rank,
270                &mut size,
271                &mut disp_unit,
272                &mut baseptr,
273            )
274        };
275        Error::check(ret)?;
276
277        let count = size as usize / std::mem::size_of::<T>();
278        // SAFETY: MPI_Win_shared_query returns a valid pointer to the shared
279        // memory segment of the specified rank. The pointer is valid for `count`
280        // elements of type T for the lifetime of the window.
281        Ok(unsafe { std::slice::from_raw_parts(baseptr.cast::<T>(), count) })
282    }
283
284    /// Fence synchronization (active target).
285    ///
286    /// A collective operation that synchronizes all accesses to the window.
287    /// All processes in the window's communicator must call this function.
288    /// After `fence()` returns, all preceding local stores are visible to
289    /// remote processes, and all remote stores are visible locally.
290    ///
291    /// This is the simplest synchronization mode, suitable for bulk-synchronous
292    /// access patterns (e.g., write phase → fence → read phase → fence).
293    ///
294    /// # Errors
295    ///
296    /// Returns an error if the MPI fence operation fails.
297    ///
298    /// # Example
299    ///
300    /// ```no_run
301    /// # use ferrompi::{Mpi, SharedWindow};
302    /// # let mpi = Mpi::init().unwrap();
303    /// # let node = mpi.world().split_shared().unwrap();
304    /// # let mut win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
305    /// // Write phase
306    /// win.local_slice_mut()[0] = 42.0;
307    ///
308    /// // Synchronize
309    /// win.fence().unwrap();
310    ///
311    /// // Read phase — all writes from all ranks are now visible
312    /// let remote = win.remote_slice(0).unwrap();
313    /// ```
314    pub fn fence(&self) -> Result<()> {
315        // SAFETY: win_handle is a valid MPI window handle.
316        let ret = unsafe { ffi::ferrompi_win_fence(0, self.win_handle) };
317        Error::check(ret)
318    }
319
320    /// Lock a specific rank's window (passive target synchronization).
321    ///
322    /// Acquires a lock on the specified rank's window, allowing one-sided
323    /// access to that rank's shared memory. Returns a [`LockGuard`] that
324    /// automatically unlocks the window when dropped.
325    ///
326    /// # Arguments
327    ///
328    /// * `lock_type` - [`LockType::Exclusive`] for read-write access, or
329    ///   [`LockType::Shared`] for read-only access
330    /// * `rank` - The rank to lock (0-based)
331    ///
332    /// # Errors
333    ///
334    /// Returns an error if the MPI lock operation fails.
335    ///
336    /// # Example
337    ///
338    /// ```no_run
339    /// # use ferrompi::{Mpi, SharedWindow, LockType};
340    /// # let mpi = Mpi::init().unwrap();
341    /// # let node = mpi.world().split_shared().unwrap();
342    /// # let win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
343    /// {
344    ///     let guard = win.lock(LockType::Shared, 0).unwrap();
345    ///     let remote = win.remote_slice(0).unwrap();
346    ///     println!("Value: {}", remote[0]);
347    ///     guard.flush().unwrap();
348    ///     // Lock is released when `guard` is dropped
349    /// }
350    /// ```
351    pub fn lock(&self, lock_type: LockType, rank: i32) -> Result<LockGuard<'_, T>> {
352        let lt = match lock_type {
353            LockType::Exclusive => ffi::FERROMPI_LOCK_EXCLUSIVE,
354            LockType::Shared => ffi::FERROMPI_LOCK_SHARED,
355        };
356        // SAFETY: win_handle is a valid MPI window handle.
357        let ret = unsafe { ffi::ferrompi_win_lock(lt, rank, 0, self.win_handle) };
358        Error::check(ret)?;
359        Ok(LockGuard { window: self, rank })
360    }
361
362    /// Lock all ranks' windows (passive target synchronization).
363    ///
364    /// Acquires shared locks on all ranks in the window's communicator.
365    /// Returns a [`LockAllGuard`] that automatically unlocks all windows
366    /// when dropped.
367    ///
368    /// This is useful for algorithms that need to access multiple ranks'
369    /// memory in a single epoch without individual lock/unlock overhead.
370    ///
371    /// # Errors
372    ///
373    /// Returns an error if the MPI lock_all operation fails.
374    ///
375    /// # Example
376    ///
377    /// ```no_run
378    /// # use ferrompi::{Mpi, SharedWindow};
379    /// # let mpi = Mpi::init().unwrap();
380    /// # let node = mpi.world().split_shared().unwrap();
381    /// # let win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
382    /// {
383    ///     let guard = win.lock_all().unwrap();
384    ///     // Access any rank's memory
385    ///     for rank in 0..node.size() {
386    ///         let remote = win.remote_slice(rank).unwrap();
387    ///         println!("Rank {rank} first value: {}", remote[0]);
388    ///     }
389    ///     guard.flush_all().unwrap();
390    ///     // All locks released when `guard` is dropped
391    /// }
392    /// ```
393    pub fn lock_all(&self) -> Result<LockAllGuard<'_, T>> {
394        // SAFETY: win_handle is a valid MPI window handle.
395        let ret = unsafe { ffi::ferrompi_win_lock_all(0, self.win_handle) };
396        Error::check(ret)?;
397        Ok(LockAllGuard { window: self })
398    }
399
400    /// Get the raw MPI window handle.
401    ///
402    /// This is provided for advanced use cases where direct access to the
403    /// underlying MPI window handle is needed (e.g., custom FFI calls).
404    pub fn raw_handle(&self) -> i32 {
405        self.win_handle
406    }
407
408    /// Get the number of processes in the window's communicator.
409    ///
410    /// This equals the size of the communicator that was used to create
411    /// the window, and determines the valid range of ranks for
412    /// [`remote_slice()`](Self::remote_slice) and [`lock()`](Self::lock).
413    pub fn comm_size(&self) -> i32 {
414        self.comm_size
415    }
416}

418impl<T: MpiDatatype> Drop for SharedWindow<T> {
419    fn drop(&mut self) {
420        // SAFETY: win_handle is a valid MPI window handle that was allocated
421        // by ferrompi_win_allocate_shared. It has not been freed yet because
422        // Drop is only called once, and we don't expose a manual free method.
423        unsafe { ffi::ferrompi_win_free(self.win_handle) };
424    }
425}

427// SharedWindow is not Send/Sync by default due to NonNull<T>.
428// This is the correct behavior — MPI windows have specific thread-safety
429// rules and should not be shared across threads without careful coordination.
430
431/// RAII guard for a single-rank window lock.
432///
433/// Created by [`SharedWindow::lock()`]. When dropped, the lock on the
434/// specified rank is automatically released via `MPI_Win_unlock`.
435///
436/// # Example
437///
438/// ```no_run
439/// # use ferrompi::{Mpi, SharedWindow, LockType};
440/// # let mpi = Mpi::init().unwrap();
441/// # let node = mpi.world().split_shared().unwrap();
442/// # let win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
443/// {
444///     let guard = win.lock(LockType::Shared, 0).unwrap();
445///     let remote = win.remote_slice(0).unwrap();
446///     // Ensure all operations are completed
447///     guard.flush().unwrap();
448/// } // Lock released here
449/// ```
450pub struct LockGuard<'a, T: MpiDatatype> {
451    window: &'a SharedWindow<T>,
452    rank: i32,
453}
454
455impl<T: MpiDatatype> LockGuard<'_, T> {
456    /// Flush pending RMA operations to the locked rank.
457    ///
458    /// Ensures that all preceding RMA operations (put, get, accumulate)
459    /// issued to the locked rank are completed at the target.
460    ///
461    /// # Errors
462    ///
463    /// Returns an error if the MPI flush operation fails.
464    pub fn flush(&self) -> Result<()> {
465        // SAFETY: The window handle is valid (borrowed from SharedWindow)
466        // and the rank was locked in the constructor.
467        let ret = unsafe { ffi::ferrompi_win_flush(self.rank, self.window.win_handle) };
468        Error::check(ret)
469    }
470}
471
472impl<T: MpiDatatype> Drop for LockGuard<'_, T> {
473    fn drop(&mut self) {
474        // SAFETY: The window handle is valid (borrowed from SharedWindow)
475        // and the rank was locked in the constructor. This unlock matches
476        // the lock call that created this guard.
477        unsafe { ffi::ferrompi_win_unlock(self.rank, self.window.win_handle) };
478    }
479}

481/// RAII guard for a lock-all window lock.
482///
483/// Created by [`SharedWindow::lock_all()`]. When dropped, all locks
484/// are automatically released via `MPI_Win_unlock_all`.
485///
486/// # Example
487///
488/// ```no_run
489/// # use ferrompi::{Mpi, SharedWindow};
490/// # let mpi = Mpi::init().unwrap();
491/// # let node = mpi.world().split_shared().unwrap();
492/// # let win = SharedWindow::<f64>::allocate(&node, 100).unwrap();
493/// {
494///     let guard = win.lock_all().unwrap();
495///     // Access shared memory from any rank
496///     guard.flush_all().unwrap();
497/// } // All locks released here
498/// ```
499pub struct LockAllGuard<'a, T: MpiDatatype> {
500    window: &'a SharedWindow<T>,
501}
502
503impl<T: MpiDatatype> LockAllGuard<'_, T> {
504    /// Flush all pending RMA operations to all ranks.
505    ///
506    /// Ensures that all preceding RMA operations issued to any rank
507    /// in the window are completed at their respective targets.
508    ///
509    /// # Errors
510    ///
511    /// Returns an error if the MPI flush_all operation fails.
512    pub fn flush_all(&self) -> Result<()> {
513        // SAFETY: The window handle is valid (borrowed from SharedWindow)
514        // and all ranks were locked in the constructor.
515        let ret = unsafe { ffi::ferrompi_win_flush_all(self.window.win_handle) };
516        Error::check(ret)
517    }
518
519    /// Flush pending RMA operations to a specific rank.
520    ///
521    /// Ensures that all preceding RMA operations issued to the specified
522    /// rank are completed at the target.
523    ///
524    /// # Arguments
525    ///
526    /// * `rank` - The rank to flush (0-based)
527    ///
528    /// # Errors
529    ///
530    /// Returns an error if the MPI flush operation fails.
531    pub fn flush(&self, rank: i32) -> Result<()> {
532        // SAFETY: The window handle is valid (borrowed from SharedWindow)
533        // and all ranks were locked in the constructor.
534        let ret = unsafe { ffi::ferrompi_win_flush(rank, self.window.win_handle) };
535        Error::check(ret)
536    }
537}
538
539impl<T: MpiDatatype> Drop for LockAllGuard<'_, T> {
540    fn drop(&mut self) {
541        // SAFETY: The window handle is valid (borrowed from SharedWindow)
542        // and all ranks were locked in the constructor. This unlock_all
543        // matches the lock_all call that created this guard.
544        unsafe { ffi::ferrompi_win_unlock_all(self.window.win_handle) };
545    }
546}
