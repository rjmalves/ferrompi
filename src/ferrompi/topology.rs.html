5/// MPI topology information gathered across all ranks in a communicator.
6///
7/// This is produced by a collective operation ([`Communicator::topology`]) and
8/// contains the rank-to-host mapping, MPI library metadata, and optional SLURM
9/// job information.
10///
11/// # Display
12///
13/// The `Display` implementation produces a human-readable topology report:
14///
15/// ```text
16/// ================ MPI Topology ================
17/// Library:   Open MPI v4.1.6
18/// Standard:  MPI 4.0
19/// Threads:   Funneled
20/// Processes: 8 across 2 nodes
21///
22///   compute-01: ranks 0, 1, 2, 3  (4 processes)
23///   compute-02: ranks 4, 5, 6, 7  (4 processes)
24/// ==============================================
25/// ```
26pub struct TopologyInfo {
27    /// MPI library version (e.g., "Open MPI v4.1.6").
28    library_version: String,
29    /// MPI standard version (e.g., "MPI 4.0").
30    standard_version: String,
31    /// Thread support level granted by the MPI runtime.
32    thread_level: ThreadLevel,
33    /// Total number of processes in the communicator.
34    size: i32,
35    /// Hosts and their assigned ranks, ordered by first rank on each host.
36    hosts: Vec<HostEntry>,
37    /// SLURM job metadata, populated when running under SLURM with the `numa` feature.
38    #[cfg(feature = "numa")]
39    slurm: Option<SlurmInfo>,
40}
41
42/// A single host and its assigned MPI ranks.
43#[derive(Debug, Clone)]
44pub struct HostEntry {
45    /// Hostname as reported by `MPI_Get_processor_name`.
46    pub hostname: String,
47    /// Sorted list of global ranks on this host.
48    pub ranks: Vec<i32>,
49}

51/// SLURM job metadata.
52#[cfg(feature = "numa")]
53#[derive(Debug, Clone)]
54pub struct SlurmInfo {

57    /// Compact node list (e.g., "node[001-004]").
58    pub node_list: Option<String>,
59    /// CPUs allocated per task.
60    pub cpus_per_task: Option<i32>,
61}

101/// Maximum hostname length used for the fixed-size allgather buffer.
102/// Matches `MPI_MAX_PROCESSOR_NAME` (256 in all major implementations).
103const HOSTNAME_BUF_LEN: usize = 256;
104
105/// Gather topology information from all ranks in the communicator.
106///
107/// This is a **collective operation** — all ranks in the communicator must call
108/// it. Every rank receives the complete topology.
109pub(crate) fn gather_topology(comm: &Communicator, mpi: &Mpi) -> Result<TopologyInfo> {
110    let size = comm.size();
111    let rank = comm.rank();
112
113    // Each rank fills a fixed-size hostname buffer.
114    let name = comm.processor_name()?;
115    let mut local_buf = [0u8; HOSTNAME_BUF_LEN];
116    let name_bytes = name.as_bytes();
117    let copy_len = name_bytes.len().min(HOSTNAME_BUF_LEN);
118    local_buf[..copy_len].copy_from_slice(&name_bytes[..copy_len]);
119
120    // Allgather the hostname buffers.
121    let mut all_bufs = vec![0u8; HOSTNAME_BUF_LEN * size as usize];
122    comm.allgather(&local_buf, &mut all_bufs)?;
123
124    // Build rank-to-host mapping, preserving insertion order (first rank seen per host).
125    let mut host_map: Vec<(String, Vec<i32>)> = Vec::new();
126    for r in 0..size {
127        let start = r as usize * HOSTNAME_BUF_LEN;
128        let end = start + HOSTNAME_BUF_LEN;
129        let raw = &all_bufs[start..end];
130        // Find the first null byte or take the whole buffer.
131        let nul_pos = raw.iter().position(|&b| b == 0).unwrap_or(HOSTNAME_BUF_LEN);
132        let hostname = std::str::from_utf8(&raw[..nul_pos])
133            .map_err(|_| Error::Internal("Invalid UTF-8 in gathered hostname".into()))?
134            .to_string();
135
136        if let Some(entry) = host_map.iter_mut().find(|(h, _)| *h == hostname) {
137            entry.1.push(r);
138        } else {
139            host_map.push((hostname, vec![r]));
140        }
141    }
142
143    let hosts: Vec<HostEntry> = host_map
144        .into_iter()
145        .map(|(hostname, ranks)| HostEntry { hostname, ranks })
146        .collect();
147
148    // Gather metadata — only rank 0 strictly needs these, but they're cheap
149    // and having them on every rank avoids conditional logic for the caller.
150    let library_version = if rank == 0 {
151        Mpi::library_version()?
152    } else {
153        String::new()
154    };
155    let standard_version = if rank == 0 {
156        Mpi::version()?
157    } else {
158        String::new()
159    };
160
161    // Broadcast the version strings from rank 0 so all ranks have them.
162    // We encode as a fixed-size buffer to keep things simple.
163    let library_version = broadcast_string(comm, &library_version, 0)?;
164    let standard_version = broadcast_string(comm, &standard_version, 0)?;
165
166    let thread_level = mpi.thread_level();
167
168    #[cfg(feature = "numa")]
169    let slurm = if crate::slurm::is_slurm_job() {
170        Some(SlurmInfo {
171            job_id: crate::slurm::job_id().unwrap_or_default(),
172            node_list: crate::slurm::node_list(),
173            cpus_per_task: crate::slurm::cpus_per_task(),
174        })
175    } else {
176        None
177    };
178
179    Ok(TopologyInfo {
180        library_version,
181        standard_version,
182        thread_level,
183        size,
184        hosts,
185        #[cfg(feature = "numa")]
186        slurm,
187    })
188}

190/// Broadcast a string from `root` to all ranks using a fixed-size buffer.
191fn broadcast_string(comm: &Communicator, s: &str, root: i32) -> Result<String> {
192    // Use a generous buffer — library version strings can be long.
193    const BUF_LEN: usize = 512;
194    let mut buf = [0u8; BUF_LEN];
195    if comm.rank() == root {
196        let bytes = s.as_bytes();
197        let copy_len = bytes.len().min(BUF_LEN);
198        buf[..copy_len].copy_from_slice(&bytes[..copy_len]);
199    }
200    comm.broadcast(&mut buf, root)?;
201    let nul_pos = buf.iter().position(|&b| b == 0).unwrap_or(BUF_LEN);
202    let result = std::str::from_utf8(&buf[..nul_pos])
203        .map_err(|_| Error::Internal("Invalid UTF-8 in broadcast string".into()))?;
204    Ok(result.to_string())
205}

207impl fmt::Display for TopologyInfo {
208    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
209        writeln!(f, "================ MPI Topology ================")?;
210        writeln!(f, "Library:   {}", self.library_version)?;
211        writeln!(f, "Standard:  {}", self.standard_version)?;
212        writeln!(f, "Threads:   {:?}", self.thread_level)?;
213        let node_word = if self.hosts.len() == 1 {
214            "node"
215        } else {
216            "nodes"
217        };
218        writeln!(
219            f,
220            "Processes: {} across {} {}",
221            self.size,
222            self.hosts.len(),
223            node_word,
224        )?;
225
226        #[cfg(feature = "numa")]
227        if let Some(ref slurm) = self.slurm {
228            writeln!(f, "SLURM Job: {}", slurm.job_id)?;
229            if let Some(ref nl) = slurm.node_list {
230                writeln!(f, "Nodes:     {}", nl)?;
231            }
232            if let Some(cpt) = slurm.cpus_per_task {
233                writeln!(f, "CPUs/Task: {}", cpt)?;
234            }
235        }
236
237        writeln!(f)?;
238        for entry in &self.hosts {
239            let ranks_str: Vec<String> = entry.ranks.iter().map(|r| r.to_string()).collect();
240            let proc_word = if entry.ranks.len() == 1 {
241                "process"
242            } else {
243                "processes"
244            };
245            writeln!(
246                f,
247                "  {}: ranks {}  ({} {})",
248                entry.hostname,
249                ranks_str.join(", "),
250                entry.ranks.len(),
251                proc_word,
252            )?;
253        }
254        write!(f, "==============================================")?;
255        Ok(())
256    }
257}
