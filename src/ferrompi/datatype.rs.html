1//! MPI datatype trait and type tag mapping.
2//!
3//! This module provides three sealed traits:
4//!
5//! - [`MpiDatatype`]: maps primitive Rust types to MPI datatype tags for all
6//!   communication operations (broadcast, send, recv, allreduce with scalar ops).
7//! - [`MpiIndexedDatatype`]: marks the six MPI paired value+index structs that are
8//!   only valid with [`MPI_MAXLOC`/`MPI_MINLOC`](crate::ReduceOp::MaxLoc) reductions.
9//! - [`BytePermutable`]: marks types safe for byte-level bitwise reductions via
10//!   `MPI_BYTE` (`BitwiseOr`, `BitwiseAnd`, `BitwiseXor`).
11//!
12//! # Primitive Types (`MpiDatatype`)
13//!
14//! | Rust Type | MPI Equivalent     | Tag Value |
15//! |-----------|-------------------|-----------|
16//! | `f32`     | `MPI_FLOAT`       | 0         |
17//! | `f64`     | `MPI_DOUBLE`      | 1         |
18//! | `i32`     | `MPI_INT32_T`     | 2         |
19//! | `i64`     | `MPI_INT64_T`     | 3         |
20//! | `u8`      | `MPI_UINT8_T`     | 4         |
21//! | `u32`     | `MPI_UINT32_T`    | 5         |
22//! | `u64`     | `MPI_UINT64_T`    | 6         |
23//!
24//! # Paired Value+Index Types (`MpiIndexedDatatype`)
25//!
26//! | Rust Struct      | MPI Equivalent        | Tag Value |
27//! |------------------|-----------------------|-----------|
28//! | [`FloatInt`]     | `MPI_FLOAT_INT`       | 7         |
29//! | [`DoubleInt`]    | `MPI_DOUBLE_INT`      | 8         |
30//! | [`LongInt`]      | `MPI_LONG_INT`        | 9         |
31//! | [`Int2`]         | `MPI_2INT`            | 10        |
32//! | [`ShortInt`]     | `MPI_SHORT_INT`       | 11        |
33//! | [`LongDoubleInt`]| `MPI_LONG_DOUBLE_INT` | 12        |
34//!
35//! # Byte-Permutable Types (`BytePermutable`)
36//!
37//! | Rust Type       | MPI Equivalent | Tag Value |
38//! |-----------------|----------------|-----------|
39//! | u8-like bytes   | `MPI_BYTE`     | 13        |
40
41/// Internal module to seal [`MpiDatatype`] — prevents external implementations.
42mod sealed {
43    pub trait Sealed {}
44}

46/// Internal module to seal [`MpiIndexedDatatype`] — a separate seal so that
47/// the two marker traits remain disjoint.  A type that implements
48/// `MpiIndexedDatatype` must NOT automatically satisfy `MpiDatatype`, and
49/// vice-versa.
50mod sealed_indexed {
51    pub trait Sealed {}
52}

54/// Internal module to seal [`BytePermutable`] — a separate seal for types
55/// whose byte representation is valid for `MPI_BYTE`-typed bitwise reductions.
56mod sealed_byte {
57    pub trait Sealed {}
58}

60/// Tag values matching C-side `FERROMPI_*` defines.
61///
62/// These discriminants must stay in sync with the `#define FERROMPI_*` values
63/// in `csrc/ferrompi.h`.
64#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
65#[repr(i32)]
66pub enum DatatypeTag {
67    /// 32-bit floating point (`MPI_FLOAT`)
68    F32 = 0,
69    /// 64-bit floating point (`MPI_DOUBLE`)
70    F64 = 1,
71    /// 32-bit signed integer (`MPI_INT32_T`)
72    I32 = 2,
73    /// 64-bit signed integer (`MPI_INT64_T`)
74    I64 = 3,
75    /// 8-bit unsigned integer (`MPI_UINT8_T`)
76    U8 = 4,
77    /// 32-bit unsigned integer (`MPI_UINT32_T`)
78    U32 = 5,
79    /// 64-bit unsigned integer (`MPI_UINT64_T`)
80    U64 = 6,
81    /// Paired `{ f32 value; i32 index }` for `MPI_FLOAT_INT` (MAXLOC/MINLOC)
82    FloatInt = 7,
83    /// Paired `{ f64 value; i32 index }` for `MPI_DOUBLE_INT` (MAXLOC/MINLOC)
84    DoubleInt = 8,
85    /// Paired `{ i64 value; i32 index }` for `MPI_LONG_INT` (MAXLOC/MINLOC).
86    /// Note: on most 64-bit platforms `long` is 8 bytes, matching `i64`.
87    LongInt = 9,
88    /// Paired `{ i32 value; i32 index }` for `MPI_2INT` (MAXLOC/MINLOC)
89    Int2 = 10,
90    /// Paired `{ i16 value; i32 index }` for `MPI_SHORT_INT` (MAXLOC/MINLOC)
91    ShortInt = 11,
92    /// Paired `{ f128-equivalent value; i32 index }` for `MPI_LONG_DOUBLE_INT`
93    /// (MAXLOC/MINLOC). Uses `[u8; 16]` on x86_64 where `long double` is 80-bit
94    /// extended precision stored in 16 bytes.
95    LongDoubleInt = 12,
96    /// Opaque 1-byte unit (`MPI_BYTE`) for type-erased bitwise reductions.
97    ///
98    /// Used exclusively with [`BytePermutable`]-bounded types and
99    /// [`Communicator::allreduce_bytes`](crate::Communicator::allreduce_bytes). The count passed to MPI is the
100    /// total byte count (`element_count * size_of::<T>()`), so MPI treats
101    /// the buffer as a flat array of bytes.
102    Byte = 13,
103}

105/// Trait for types that can be used in MPI communication operations.
106///
107/// This is a **sealed trait** — it cannot be implemented outside this crate.
108/// Supported types: [`f32`], [`f64`], [`i32`], [`i64`], [`u8`], [`u32`], [`u64`].
109///
110/// # Example
111///
112/// ```no_run
113/// use ferrompi::{Mpi, MpiDatatype};
114///
115/// let mpi = Mpi::init().unwrap();
116/// let world = mpi.world();
117///
118/// // Works with f64
119/// let mut data_f64 = vec![1.0f64; 10];
120/// world.broadcast(&mut data_f64, 0).unwrap();
121///
122/// // Works with i32
123/// let mut data_i32 = vec![42i32; 10];
124/// world.broadcast(&mut data_i32, 0).unwrap();
125/// ```
126pub trait MpiDatatype: sealed::Sealed + Copy + Send + 'static {
127    /// The datatype tag used for FFI dispatch to the C layer.
128    const TAG: DatatypeTag;
129}
130
131macro_rules! impl_mpi_datatype {
132    ($ty:ty, $tag:expr) => {
133        impl sealed::Sealed for $ty {}
134        impl MpiDatatype for $ty {
135            const TAG: DatatypeTag = $tag;
136        }
137    };
138}

140impl_mpi_datatype!(f32, DatatypeTag::F32);
141impl_mpi_datatype!(f64, DatatypeTag::F64);
142impl_mpi_datatype!(i32, DatatypeTag::I32);
143impl_mpi_datatype!(i64, DatatypeTag::I64);
144impl_mpi_datatype!(u8, DatatypeTag::U8);
145impl_mpi_datatype!(u32, DatatypeTag::U32);
146impl_mpi_datatype!(u64, DatatypeTag::U64);

152/// Trait for types that can be used with `MPI_MAXLOC` / `MPI_MINLOC`.
153///
154/// This is a **sealed trait** — it cannot be implemented outside this crate.
155/// Only the six MPI predefined paired value+index types implement it.
156///
157/// Use these types exclusively with [`Communicator::allreduce_indexed`](crate::Communicator::allreduce_indexed) and
158/// [`ReduceOp::MaxLoc`](crate::ReduceOp::MaxLoc) / [`ReduceOp::MinLoc`](crate::ReduceOp::MinLoc). They are **not** valid for
159/// `broadcast`, `send`, `recv`, or other collectives (MPI treats them as
160/// opaque structure types that require `MPI_Type_commit`; ferrompi does not
161/// yet manage committed derived types — that is Epic 6).
162///
163/// # Example
164///
165/// ```no_run
166/// use ferrompi::{Mpi, ReduceOp, DoubleInt, MpiIndexedDatatype};
167///
168/// let mpi = Mpi::init().unwrap();
169/// let world = mpi.world();
170/// let rank = world.rank();
171///
172/// let send = [DoubleInt { value: rank as f64, index: rank }];
173/// let mut recv = [DoubleInt { value: 0.0, index: 0 }];
174/// world.allreduce_indexed(&send, &mut recv, ReduceOp::MaxLoc).unwrap();
175/// // Every rank now holds { value: (size-1) as f64, index: size-1 }
176/// ```
177///
178/// # Sealed: cannot implement for external types
179///
180/// ```compile_fail
181/// use ferrompi::MpiIndexedDatatype;
182/// use ferrompi::DatatypeTag;
183///
184/// // This must not compile — MpiIndexedDatatype is sealed.
185/// impl MpiIndexedDatatype for i32 {
186///     const TAG: DatatypeTag = DatatypeTag::Int2;
187/// }
188/// ```
189pub trait MpiIndexedDatatype: sealed_indexed::Sealed + Copy + Send + 'static {
190    /// The datatype tag used for FFI dispatch to the C layer.
191    const TAG: DatatypeTag;
192}

194/// Paired `{ f32 value; i32 index }` — maps to `MPI_FLOAT_INT`.
195///
196/// The `index` field conventionally holds the rank of the contributing process.
197/// Use with [`ReduceOp::MaxLoc`](crate::ReduceOp::MaxLoc) or [`ReduceOp::MinLoc`](crate::ReduceOp::MinLoc) via
198/// [`Communicator::allreduce_indexed`](crate::Communicator::allreduce_indexed)(crate::Communicator::allreduce_indexed).
199#[derive(Debug, Clone, Copy, PartialEq)]
200#[repr(C)]
201pub struct FloatInt {
202    /// The floating-point value to reduce.
203    pub value: f32,
204    /// The index (rank) associated with this value.
205    pub index: i32,
206}

208/// Paired `{ f64 value; i32 index }` — maps to `MPI_DOUBLE_INT`.
209///
210/// The `index` field conventionally holds the rank of the contributing process.
211/// Use with [`ReduceOp::MaxLoc`](crate::ReduceOp::MaxLoc) or [`ReduceOp::MinLoc`](crate::ReduceOp::MinLoc) via
212/// [`Communicator::allreduce_indexed`](crate::Communicator::allreduce_indexed)(crate::Communicator::allreduce_indexed).
213///
214/// Layout on 64-bit Linux: `sizeof == 16`, `alignof == 8` (4 bytes of trailing
215/// padding after the `i32` index to satisfy the `f64` alignment of the next
216/// element in an array).
217#[derive(Debug, Clone, Copy, PartialEq)]
218#[repr(C)]
219pub struct DoubleInt {
220    /// The double-precision value to reduce.
221    pub value: f64,
222    /// The index (rank) associated with this value.
223    pub index: i32,
224}

276/// Paired `{ long double value; i32 index }` — maps to `MPI_LONG_DOUBLE_INT`.
277///
278/// C `long double` on x86_64 Linux is 80-bit extended precision stored in
279/// 16 bytes (with 6 bytes of padding). Rust has no native `f80` type, so
280/// the value field is stored as `[u8; 16]` to match the C struct layout
281/// exactly. Users working with this type should cast to/from the actual
282/// `long double` representation via FFI if needed.
283///
284/// The `index` field conventionally holds the rank of the contributing process.
285/// Use with [`ReduceOp::MaxLoc`](crate::ReduceOp::MaxLoc) or [`ReduceOp::MinLoc`](crate::ReduceOp::MinLoc) via
286/// [`Communicator::allreduce_indexed`](crate::Communicator::allreduce_indexed)(crate::Communicator::allreduce_indexed).
287///
288/// Layout on x86_64 Linux: `sizeof == 32`, `alignof == 16`.
289/// Layout on aarch64 Linux: `sizeof == 32`, `alignof == 16`
290/// (128-bit quad-precision `long double`, 16 bytes value + 4-byte index +
291/// 12 bytes trailing padding).
292/// On x86_64 Linux, `long double` has 16-byte alignment; use `repr(C, align(16))`
293/// so that the Rust struct layout matches the C struct layout exactly.
294/// On other platforms this over-aligns harmlessly (MPI will still accept it).
295#[derive(Debug, Clone, Copy, PartialEq)]
296#[repr(C, align(16))]
297pub struct LongDoubleInt {
298    /// Raw bytes of the `long double` value (platform-specific layout).
299    pub value: [u8; 16],
300    /// The index (rank) associated with this value.
301    pub index: i32,
302}

324/// Trait for types whose byte representation is valid for use with
325/// `MPI_BYTE`-typed bitwise reductions.
326///
327/// This is a **sealed trait** — only types whose memory layout
328/// consists entirely of meaningful bytes (no padding, no
329/// discriminants, no uninhabited cases) can implement it. ferrompi
330/// provides blanket impls for `u8`, `u16`, `u32`, `u64`, `i8`,
331/// `i16`, `i32`, `i64`, plus `[T; N]` where `T: BytePermutable`.
332///
333/// # Safety invariants (enforced by sealing)
334///
335/// - No padding bytes inside the type (every byte is meaningful).
336/// - No niche optimizations: every bit pattern is a valid value.
337/// - `Copy + Send + 'static`.
338///
339/// # Sealed: cannot implement for external types
340///
341/// ```compile_fail
342/// use ferrompi::BytePermutable;
343///
344/// // This must not compile — BytePermutable is sealed.
345/// impl BytePermutable for f64 {}
346/// ```
347pub trait BytePermutable: sealed_byte::Sealed + Copy + Send + 'static {}
348
349macro_rules! impl_byte_permutable {
350    ($ty:ty) => {
351        impl sealed_byte::Sealed for $ty {}
352        impl BytePermutable for $ty {}
353    };
354}
355
356impl_byte_permutable!(u8);
357impl_byte_permutable!(u16);
358impl_byte_permutable!(u32);
359impl_byte_permutable!(u64);
360impl_byte_permutable!(i8);
361impl_byte_permutable!(i16);
362impl_byte_permutable!(i32);
363impl_byte_permutable!(i64);
364
365impl<T: BytePermutable, const N: usize> sealed_byte::Sealed for [T; N] {}
366impl<T: BytePermutable, const N: usize> BytePermutable for [T; N] {}

545        assert_eq!(align_of::<Int2>(), 4, "Int2 align");

547        // ShortInt: { i16, i32 } — 2 bytes padding between fields for i32 alignment
548        // sizeof == 8, alignof == 4
549        assert!(
550            size_of::<ShortInt>() >= 6,
551            "ShortInt must hold at least i16 + i32"
552        );
553        assert!(
554            align_of::<ShortInt>() >= 4,
555            "ShortInt alignment must be at least i32 alignment"
556        );
557        #[cfg(any(target_arch = "x86_64", target_arch = "aarch64"))]
558        {
559            assert_eq!(size_of::<ShortInt>(), 8, "ShortInt size on x86_64/aarch64");
560            assert_eq!(
561                align_of::<ShortInt>(),
562                4,
563                "ShortInt align on x86_64/aarch64"
564            );
565        }
566
567        // LongDoubleInt: { [u8;16], i32 } — value is 16 bytes, index is 4 bytes,
568        // trailing padding to satisfy alignment. On x86_64: sizeof == 20 rounds up
569        // to 32 due to 16-byte alignment of long double. On aarch64: sizeof == 32.
570        assert!(
571            size_of::<LongDoubleInt>() >= 20,
572            "LongDoubleInt must hold at least [u8;16] + i32"
573        );
574        assert!(
575            align_of::<LongDoubleInt>() >= 1,
576            "LongDoubleInt must have at least 1-byte alignment"
577        );
578        // On x86_64 Linux, MPI_LONG_DOUBLE_INT is { long double (16 bytes), int (4 bytes) }
579        // with 12 bytes trailing padding, total 32 bytes, aligned to 16 bytes.
580        #[cfg(target_arch = "x86_64")]
581        {
582            assert_eq!(
583                size_of::<LongDoubleInt>(),
584                32,
585                "LongDoubleInt size on x86_64"
586            );
587            assert_eq!(
588                align_of::<LongDoubleInt>(),
589                16,
590                "LongDoubleInt align on x86_64"
591            );
592        }
593    }
