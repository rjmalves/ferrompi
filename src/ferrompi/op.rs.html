1//! User-defined MPI reduction operations via [`MPI_Op_create`].
2//!
3//! This module provides [`UserOp<T>`], a safe wrapper around a user-supplied
4//! Rust closure that MPI invokes during reduction collectives
5//! (`MPI_Reduce`, `MPI_Allreduce`, `MPI_Scan`, etc.).
6//!
7//! ## Safety model
8//!
9//! The closure must satisfy `Send + Sync + 'static`:
10//!
11//! * **`Send`** — the closure is moved into a static slot table accessible from
12//!   any thread (MPI may call it from an internal thread pool).
13//! * **`Sync`** — under `MPI_THREAD_MULTIPLE`, the same op may be invoked
14//!   concurrently from several threads.
15//! * **`'static`** — the closure is held until `MPI_Op_free` returns, which may
16//!   be much later than the call site.
17//!
18//! ## Panic behaviour
19//!
20//! **A panic inside the closure aborts the process immediately.**
21//!
22//! Panicking across a C FFI boundary is undefined behaviour; there is no
23//! mechanism for MPI to propagate or observe a Rust panic.  The trampoline
24//! wraps every closure call in [`std::panic::catch_unwind`]: on `Err` it calls
25//! [`std::process::abort`] before the panic can reach the C frame.  Treat a
26//! panic inside a reduction closure as a fatal programming error.
27//!
28//! ## Slot-table limit
29//!
30//! The implementation supports at most **16** concurrently live `UserOp`
31//! instances per process.  Attempting to create a seventeenth returns
32//! [`Error::Mpi`] with class `Other`.
33//!
34//! ## `compile_fail` doctest — `Send + Sync` bound
35//!
36//! ```compile_fail
37//! use ferrompi::UserOp;
38//! let local = std::rc::Rc::new(0i32);
39//! let op = UserOp::new(move |_a: &[f64], _b: &mut [f64]| {
40//!     let _ = local.clone();
41//! });
42//! ```
43

51// ============================================================================
52// Slot count — must match MAX_OPS in csrc/ferrompi.c
53// ============================================================================
54const MAX_OPS: usize = 16;
55
56/// Type alias for the byte-level closure stored in each op slot.
57///
58/// Using an alias avoids the `clippy::type_complexity` lint at every use site.
59type ByteClosure = Box<dyn Fn(&[u8], &mut [u8]) + Send + Sync + 'static>;
60
61// ============================================================================
62// Static closure registry
63//
64// Each slot holds a byte-level closure adapter.  The adapter is a
65// `Box<dyn Fn(&[u8], &mut [u8]) + Send + Sync + 'static>` constructed in
66// `UserOp::new_impl` from the caller's typed `Fn(&[T], &mut [T])` by wrapping
67// it in a byte-reinterpreting adapter closure.
68//
69// OnceLock is used so that each slot can be written exactly once and read many
70// times concurrently, without requiring a Mutex.  Dropping the UserOp must
71// reconstruct the Box from the raw pointer rather than going through OnceLock
72// again (OnceLock does not expose a reset path); see `ferrompi_op_drop_closure`.
73// ============================================================================
74
75/// Per-slot registry of raw fat-pointer halves.
76///
77/// We cannot store `Box<dyn Fn(...)>` in a static array of `OnceLock` because
78/// statics require `const`-initializable values and `OnceLock::new()` is
79/// `const`-stable only from Rust 1.70+, but more importantly we need to be
80/// able to reconstruct and drop the `Box` from `ferrompi_op_drop_closure`,
81/// which is called from C and cannot go through the OnceLock API.
82///
83/// The chosen approach: store the two halves of the fat pointer (`*mut ()` data
84/// and `*mut ()` vtable) as atomic raw pointers.  We encode "unset" as null and
85/// "set" as non-null.  The fat pointer is stored atomically so that concurrent
86/// reads from MPI threads are data-race-free (Relaxed load is sufficient since
87/// the store in `new_impl` happens-before any MPI invocation of the trampoline —
88/// MPI_Op_create establishes that ordering).
89///
90/// In practice the registry data and vtable are written once (in `new_impl`)
91/// and then only read (in `rust_user_op_invoke`) or reset to null (in
92/// `ferrompi_op_drop_closure`). We use `AtomicPtr` with `Relaxed` ordering
93/// because:
94///   * The store in `new_impl` is sequenced before `MPI_Op_create` which is the
95///     happens-before anchor for all subsequent MPI trampoline calls.
96///   * `ferrompi_op_drop_closure` is called from `ferrompi_op_free` which is
97///     called only after `MPI_Op_free` returns — MPI guarantees no further
98///     trampoline calls after that point.
99use std::sync::atomic::{AtomicPtr, Ordering};
100
101struct RegistrySlot {
102    data: AtomicPtr<()>,
103    vtbl: AtomicPtr<()>,
104}
105
106impl RegistrySlot {
107    const fn new() -> Self {
108        Self {
109            data: AtomicPtr::new(std::ptr::null_mut()),
110            vtbl: AtomicPtr::new(std::ptr::null_mut()),
111        }
112    }
113}

115// SAFETY: AtomicPtr<()> is Send + Sync by design; raw pointers are wrapped in
116// atomics which provide the necessary synchronisation.
117unsafe impl Send for RegistrySlot {}
118unsafe impl Sync for RegistrySlot {}

121    RegistrySlot::new(),
122    RegistrySlot::new(),
123    RegistrySlot::new(),
124    RegistrySlot::new(),

143/// Called by each C trampoline `ferrompi_user_op_trampoline_N`.
144///
145/// The C trampoline passes the slot's fat-pointer halves and the raw buffer
146/// pointers it received from MPI.  This function reconstructs the byte-level
147/// closure and invokes it, wrapped in `catch_unwind`.
148///
149/// # Safety
150///
151/// * `closure_data` and `closure_vtbl` are the two halves of a valid
152///   `*mut dyn Fn(&[u8], &mut [u8]) + Send + Sync + 'static` fat pointer
153///   previously stored by `UserOp::new_impl`.
154/// * `invec` is a valid read-only pointer to `len * byte_size` bytes.
155/// * `inoutvec` is a valid read-write pointer to `len * byte_size` bytes.
156/// * `dt_tag` is the `FERROMPI_*` tag that matches the type `T` the `UserOp`
157///   was parameterised with.
158///
159/// Called from C, so the ABI must be exactly `extern "C"`.
160#[unsafe(no_mangle)]
161pub unsafe extern "C" fn rust_user_op_invoke(
162    closure_data: *mut c_void,
163    closure_vtbl: *mut c_void,
164    invec: *const c_void,
165    inoutvec: *mut c_void,
166    len: std::ffi::c_int,
167    _dt_tag: std::ffi::c_int,
168) {
169    // Reconstruct the fat pointer from its two halves.
170    // The fat pointer layout is [data_ptr, vtable_ptr] on all Rust targets;
171    // we encode it as a 2-element array of *mut () and transmute.
172    let fat_ptr_halves: [*mut (); 2] = [closure_data.cast(), closure_vtbl.cast()];
173    // SAFETY: fat_ptr_halves holds a valid fat pointer for
174    // `*const dyn Fn(&[u8], &mut [u8]) + Send + Sync`.  The memory it points
175    // to is alive for the duration of this call: ferrompi_op_free (which frees
176    // the closure) calls MPI_Op_free first and only drops the closure after
177    // MPI_Op_free returns, so no concurrent drop can occur here.
178    // SAFETY: fat_ptr_halves encodes a valid &dyn for the ByteClosure stored
179    // at slot registration time.  We borrow it as a shared reference; the
180    // Box is still owned (it is dropped only in ferrompi_op_drop_closure, after
181    // MPI_Op_free returns).
182    let closure: &(dyn Fn(&[u8], &mut [u8]) + Send + Sync) =
183        // Transmute [*mut (); 2] → fat-pointer reference.  This is the standard
184        // pattern for reconstructing a dyn reference from a stored fat pointer.
185        unsafe { std::mem::transmute(fat_ptr_halves) };
186
187    // Build byte slices from the raw MPI buffers.
188    // len is the number of *elements* (MPI's *len parameter).  The byte-level
189    // adapter stored in the registry receives slices whose .len() is the
190    // element count — it uses that to reconstruct typed &[T] / &mut [T] slices
191    // of the correct length via slice::from_raw_parts.
192    //
193    // We do NOT multiply by size_of::<T>() here; that knowledge lives entirely
194    // inside the adapter closure captured in UserOp::new_impl.  Passing the
195    // element count as the u8-slice length avoids any accidental OOB: the
196    // adapter must not interpret .len() as a byte count.
197    let len_usize = len as usize;
198    // SAFETY: Both slices span `len * size_of::<T>()` bytes at the MPI-provided
199    // addresses; the adapter casts the pointer and uses len_usize as the element
200    // count to reconstruct properly-typed slices.  The adapter must not use
201    // .len() as a byte count — it is the MPI element count.
202    let invec_bytes: &[u8] = unsafe { std::slice::from_raw_parts(invec.cast::<u8>(), len_usize) };
203    let inoutvec_bytes: &mut [u8] =
204        unsafe { std::slice::from_raw_parts_mut(inoutvec.cast::<u8>(), len_usize) };
205
206    // Wrap the closure call in catch_unwind (ADR-0005 Decision 6).
207    // A panic across the FFI boundary is UB; abort on Err.
208    let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
209        closure(invec_bytes, inoutvec_bytes);
210    }));
211    if result.is_err() {
212        // A panic inside a user-defined reduction closure is a fatal
213        // programming error.  Abort immediately to prevent silent data
214        // corruption in the collective result (ADR-0005 Decision 6).
215        std::process::abort();
216    }
217}

219/// Called by `ferrompi_op_free` (in C) after `MPI_Op_free` returns.
220///
221/// Reconstructs the `Box<dyn Fn(...)>` from the raw fat pointer halves stored
222/// in the slot and drops it.  After this function returns, the slot is cleared
223/// by `free_op_slot` in C.
224///
225/// # Safety
226///
227/// * `slot` must be in range `0..MAX_OPS`.
228/// * The fat pointer stored in `REGISTRY[slot]` must point to a valid
229///   `Box<dyn Fn(&[u8], &mut [u8]) + Send + Sync + 'static>` that was
230///   previously stored by `UserOp::new_impl` and has not yet been dropped.
231/// * This function is called exactly once per `UserOp`, from C, after
232///   `MPI_Op_free` has returned.
233#[unsafe(no_mangle)]
234pub unsafe extern "C" fn ferrompi_op_drop_closure(slot: i32) {
235    if slot < 0 || slot as usize >= MAX_OPS {
236        return;
237    }
238    let idx = slot as usize;
239    let data = REGISTRY[idx]
240        .data
241        .swap(std::ptr::null_mut(), Ordering::Relaxed);
242    let vtbl = REGISTRY[idx]
243        .vtbl
244        .swap(std::ptr::null_mut(), Ordering::Relaxed);
245    if data.is_null() || vtbl.is_null() {
246        return;
247    }
248    // Reconstruct the fat pointer and drop the Box.
249    let fat_ptr_halves: [*mut (); 2] = [data, vtbl];
250    // SAFETY: fat_ptr_halves is the valid fat pointer stored by new_impl.
251    // We are reconstructing ownership of the Box; it has not been dropped.
252    // MPI_Op_free has returned so no trampoline call can race with this drop.
253    let closure: ByteClosure = unsafe { std::mem::transmute(fat_ptr_halves) };
254    drop(closure);
255}

261/// A user-defined MPI reduction operation backed by a Rust closure.
262///
263/// `T` must implement [`MpiDatatype`] — i.e., it must be one of the primitive
264/// types recognised by ferrompi (`f32`, `f64`, `i32`, `i64`, `u8`, `u32`,
265/// `u64`).
266///
267/// The closure receives `invec` as a shared slice and `inoutvec` as a mutable
268/// slice of the same length.  It must accumulate `invec[i]` into `inoutvec[i]`
269/// for each index `i` — the standard MPI semantics for user reduction
270/// functions.
271///
272/// # Thread-safety
273///
274/// The closure is called from whichever thread MPI uses internally for the
275/// reduction.  Under `MPI_THREAD_MULTIPLE` the same op may be invoked
276/// concurrently from multiple threads; the closure must be safe for concurrent
277/// invocation, enforced by the `Sync` bound.
278///
279/// # Panic behaviour
280///
281/// A panic inside the closure **aborts the process**.  See module-level
282/// documentation for details.
283///
284/// # Slot-table limit
285///
286/// At most 16 `UserOp` instances may be live concurrently per process.
287///
288/// # Examples
289///
290/// ```no_run
291/// use ferrompi::{Mpi, UserOp};
292///
293/// let mpi = Mpi::init().unwrap();
294/// let world = mpi.world();
295///
296/// let op: UserOp<f64> = UserOp::new(|invec: &[f64], inoutvec: &mut [f64]| {
297///     for (x, y) in invec.iter().zip(inoutvec.iter_mut()) {
298///         *y = x.max(*y);
299///     }
300/// }).unwrap();
301///
302/// let send = vec![world.rank() as f64 + 1.5_f64];
303/// let mut recv = vec![0.0_f64];
304/// world.allreduce_with_op(&send, &mut recv, &op).unwrap();
305/// ```
306pub struct UserOp<T: MpiDatatype> {
307    handle: i32,
308    _marker: PhantomData<T>,
309}
310
311impl<T: MpiDatatype> UserOp<T> {
312    /// Create a commutative user-defined reduction op.
313    ///
314    /// MPI is permitted to reorder operands for optimisation purposes.  Use
315    /// this constructor for operations that satisfy `f(a, b) == f(b, a)` —
316    /// element-wise sums, maxima, minima, etc.
317    ///
318    /// # Errors
319    ///
320    /// Returns `Err` if the op-slot table is full (16 concurrent `UserOp`s)
321    /// or if `MPI_Op_create` fails.
322    pub fn new<F>(f: F) -> Result<Self>
323    where
324        F: Fn(&[T], &mut [T]) + Send + Sync + 'static,
325    {
326        Self::new_impl(f, 1)
327    }
328
329    /// Create a non-commutative user-defined reduction op.
330    ///
331    /// MPI will not reorder operands.  Use this constructor for operations
332    /// where order matters — matrix multiplication, string concatenation, etc.
333    ///
334    /// # Errors
335    ///
336    /// Returns `Err` if the op-slot table is full or if `MPI_Op_create` fails.
337    pub fn new_noncommutative<F>(f: F) -> Result<Self>
338    where
339        F: Fn(&[T], &mut [T]) + Send + Sync + 'static,
340    {
341        Self::new_impl(f, 0)
342    }
343
344    fn new_impl<F>(f: F, commute: i32) -> Result<Self>
345    where
346        F: Fn(&[T], &mut [T]) + Send + Sync + 'static,
347    {
348        // Step 1: allocate a slot in the C-side op table.
349        let mut slot: i32 = -1;
350        let ret = unsafe { ffi::ferrompi_op_alloc_slot(&mut slot) };
351        if ret != 0 {
352            return Err(Error::Mpi {
353                class: MpiErrorClass::Other,
354                code: ret,
355                message: "op-slot table is full (MAX_OPS=16 concurrent UserOps)".to_string(),
356                operation: Some("op_create"),
357            });
358        }
359        let idx = slot as usize;
360        debug_assert!(idx < MAX_OPS, "slot out of range");
361
362        // Step 2: wrap the typed closure in a byte-level adapter.
363        //
364        // The C trampoline passes byte slices whose .len() field carries the
365        // MPI element count (not byte count).  The adapter uses that element
366        // count directly to reconstruct typed slices via slice::from_raw_parts.
367        //
368        // Why byte-level: the Rust callback `rust_user_op_invoke` has a single
369        // signature regardless of T; it reconstructs a
370        // `dyn Fn(&[u8], &mut [u8])` trait object.  The adapter converts back
371        // to `&[T]` / `&mut [T]` via `slice::from_raw_parts`, interpreting
372        // .len() as the element count (not a byte count).
373        // Wrap in a ByteClosure (byte-level adapter over the typed closure).
374        let byte_closure: ByteClosure =
375            Box::new(move |invec_bytes: &[u8], inoutvec_bytes: &mut [u8]| {
376                // `invec_bytes.len()` and `inoutvec_bytes.len()` are the MPI
377                // element count forwarded by rust_user_op_invoke.  The actual
378                // byte span is elem_count * size_of::<T>(), which MPI guarantees
379                // is valid; we use elem_count here as the slice element count.
380                let elem_count = invec_bytes.len();
381                // SAFETY:
382                //   * invec_bytes.as_ptr() points to a valid MPI-provided buffer
383                //     of at least elem_count * size_of::<T>() bytes.
384                //   * T: MpiDatatype implies T: Copy with stable layout; MPI
385                //     provides properly-aligned buffers for the registered type.
386                //   * elem_count comes from MPI's *len — the number of elements
387                //     MPI needs reduced.
388                //   * .len() is used here as element count, NOT byte count.
389                let invec: &[T] = unsafe {
390                    std::slice::from_raw_parts(invec_bytes.as_ptr().cast::<T>(), elem_count)
391                };
392                let inoutvec: &mut [T] = unsafe {
393                    std::slice::from_raw_parts_mut(
394                        inoutvec_bytes.as_mut_ptr().cast::<T>(),
395                        elem_count,
396                    )
397                };
398                f(invec, inoutvec);
399            });
400
401        // Step 3: store the fat pointer in the static registry.
402        //
403        // We decompose the Box into its two fat-pointer halves and store them
404        // as raw pointers so that the C layer can pass them back via
405        // rust_user_op_invoke, and so that ferrompi_op_drop_closure can
406        // reconstruct and drop the Box.
407        let raw_fat: [*mut (); 2] = unsafe {
408            // SAFETY: transmuting Box<dyn Fn(...)> into [*mut (); 2] extracts
409            // the fat pointer halves without running the destructor.  We
410            // reconstruct the Box in ferrompi_op_drop_closure.
411            std::mem::transmute(Box::into_raw(byte_closure))
412        };
413        // Release store: subsequent loads in the trampoline (Relaxed) are
414        // guaranteed to observe these values because the call to
415        // ferrompi_op_create_user (MPI_Op_create) establishes the
416        // happens-before edge between this store and any trampoline invocation.
417        REGISTRY[idx].data.store(raw_fat[0], Ordering::Release);
418        REGISTRY[idx].vtbl.store(raw_fat[1], Ordering::Release);
419
420        // Step 4: call MPI_Op_create via the C shim.
421        //
422        // ferrompi_op_set_closure stores the fat-pointer halves into the C-side
423        // op_closure_data/op_closure_vtbl arrays so the C trampolines can pass
424        // them to rust_user_op_invoke.
425        unsafe {
426            ffi::ferrompi_op_set_closure(slot, raw_fat[0].cast(), raw_fat[1].cast());
427        }
428
429        let mut handle: i32 = -1;
430        let ret = unsafe { ffi::ferrompi_op_create_user(slot, commute, &mut handle) };
431        if ret != 0 {
432            // Rollback: MPI_Op_create failed so no MPI_Op was registered.
433            // We must NOT call ferrompi_op_free here — that would call
434            // MPI_Op_free on MPI_OP_NULL (the slot was never populated),
435            // which is implementation-defined behaviour.
436            //
437            // Instead:
438            //   1. Reconstruct and drop the Box<dyn Fn(...)> directly from
439            //      the registry fat-pointer halves we stored above.
440            //   2. Call ferrompi_op_free_slot_only to clear the closure
441            //      pointers and release the op_used slot without touching
442            //      MPI_Op_free.
443            let data = REGISTRY[idx]
444                .data
445                .swap(std::ptr::null_mut(), Ordering::Relaxed);
446            let vtbl = REGISTRY[idx]
447                .vtbl
448                .swap(std::ptr::null_mut(), Ordering::Relaxed);
449            if !data.is_null() && !vtbl.is_null() {
450                let fat: [*mut (); 2] = [data, vtbl];
451                // SAFETY: we just stored this value above; it has not been
452                // dropped yet and this is the only owner.
453                let closure: ByteClosure = unsafe { std::mem::transmute(fat) };
454                drop(closure);
455            }
456            // Release the C-side slot without calling MPI_Op_free.
457            unsafe { ffi::ferrompi_op_free_slot_only(slot) };
458            return Err(Error::from_code_with_op(ret, "op_create"));
459        }
460        debug_assert_eq!(handle, slot);
461
462        Ok(UserOp {
463            handle,
464            _marker: PhantomData,
465        })
466    }
467
468    /// Return the raw slot handle (for use by `allreduce_with_op`).
469    #[inline]
470    pub(crate) fn raw_handle(&self) -> i32 {
471        self.handle
472    }
473}
474
475impl<T: MpiDatatype> Drop for UserOp<T> {
476    fn drop(&mut self) {
477        // Drop ordering (ADR-0005 Decision 3):
478        //   1. ferrompi_op_free → MPI_Op_free (MPI will not invoke the
479        //      trampoline after this returns).
480        //   2. ferrompi_op_free → ferrompi_op_drop_closure (Rust callback,
481        //      drops the Box).
482        //   3. ferrompi_op_free → free_op_slot (reclaims the C-side slot).
483        //
484        // The handle is valid for the lifetime of this UserOp; it is freed
485        // exactly once here.
486        let ret = unsafe {
487            // SAFETY: self.handle was allocated by UserOp::new_impl and has
488            // not been freed.  Drop is called exactly once.
489            ffi::ferrompi_op_free(self.handle)
490        };
491        // Log but do not panic in Drop.
492        if ret != 0 {
493            eprintln!("ferrompi: UserOp::drop — ferrompi_op_free returned error code {ret}");
494        }
495    }
496}
